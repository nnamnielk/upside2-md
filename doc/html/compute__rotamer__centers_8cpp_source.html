24static float dihedral(
25 Vector3f r1, Vector3f r2, Vector3f r3, Vector3f r4)
26 // Formulas and notation taken from Blondel and Karplus, 1995
27{
28 Vector3f F = r1-r2;
29 Vector3f G = r2-r3;
30 Vector3f H = r4-r3;
31
32 Vector3f A = F.cross(G);
33 Vector3f B = H.cross(G);
34 Vector3f C = B.cross(A);
35
36 return atan2(C.dot(G), A.dot(B) * G.norm());
37}

159string np_string(const MatrixX3f &x) {
160 // WARNING not efficient for large matrices because of the string concatenation
161 string s = "";
162 for(int nr=0; nr<x.rows(); ++nr) {
163 string ln = nr ? " [" : "np.array([[";
164 for(int nc=0; nc<x.cols(); ++nc)
165 ln += to_string(x(nr,nc)) + ", ";
166 ln += (nr<x.rows()-1) ? "],\n" : "]])\n";
167 s += ln;
168 }
169 return s;
170}

172MatrixXf dmat(MatrixX3f& x) {
173 int n=x.rows();
174 MatrixXf ret(n, n);
175 for(int i=0; i<n; ++i) for(int j=0; j<n; ++j) ret(i,j) = (x.row(i)-x.row(j)).norm();
176 return ret;
177}

180float angle_wrap(float x) {
181 while(x> M_PI_F) x-=2.f*M_PI_F;
182 while(x<-M_PI_F) x+=2.f*M_PI_F;
183 return x;
184}

308int get_beads(const std::string& rname, vector<float>& centers, vector<float>& direcs, MatrixX3f &pos) {
309 int n_bead = 0;
310 auto pc = [&](const RowVector3f& v) {push_vector(centers, v); n_bead++;};
311 auto pd = [&](const RowVector3f& v) {push_vector(direcs, v.normalized());};
312
313 if(rname == "GLY") {
314 pc(pos.row(1));
315 pd(pos.row(1) - 0.5f*(pos.row(0) + pos.row(2)));
316 } else if(rname == "ALA"){
317 pc(pos.row(4));
318 pd(pos.row(4) - pos.row(1));
319 } else if(rname == "GLU") {
320 pc(pos.bottomRows(pos.rows()-6).colwise().mean());
321 pd(pos.row(5)-pos.row(4)); // CB->CG bond vector
322 } else if(rname == "LYS") {
323 pc(pos.row(8)); // NZ atom
324 pc(pos.row(6)); // CD atom
325 pd(pos.row(5)-pos.row(4)); // CB->CG bond vector
326 pd(pos.row(5)-pos.row(4)); // CB->CG bond vector
327 } else if(rname == "ARG") {
328 pc((1.f/3.f)*(pos.row(7)+pos.row(9)+pos.row(10))); // center of nitrogens
329 pd(pos.row(5)-pos.row(4)); // CB->CG bond vector
330 } else if(rname == "ASP") {
331 pc(pos.row(5)); // CG atom
332 pd(pos.row(5)-pos.row(4)); // CB->CG bond vector
333 } else if(rname == "THR") {
334 pc(pos.row(5)); // atom 5 is OG1
335 pc(pos.row(6)); // atom 6 is CG1
336 pd(pos.row(5)-pos.row(4)); // CB->OG1 bond vector
337 pd(pos.row(6)-pos.row(4)); // CB->CG1 bond vector
338 } else if(rname == "TRP") {
339 pc((1.f/6.f)*(pos.row(7)+pos.row(9)+pos.row(10)+pos.row(11)+pos.row(12)+pos.row(13))); // 2nd aromatic ring
340 pc(pos.row(8)); // atom 8 is NE1
341 pd(pos.row(5)-pos.row(4)); // CB->CG bond vector
342 pd(pos.row(8)-pos.row(4)); // CB->NE1 bond vector
343 } else if(rname == "TYR") {
344 pc((1.f/6.f)*(pos.row(5)+pos.row(6)+pos.row(7)+pos.row(8)+pos.row(9)+pos.row(10))); // aromatic ring
345 pc(pos.row(11)); // atom 11 is OH
346 pd(pos.row( 5)-pos.row(4)); // CB->CG bond vector
347 pd(pos.row(11)-pos.row(4)); // CB->OH bond vector
348 } else {
349 RowVector3f rest = pos.bottomRows(pos.rows()-5).colwise().mean();
350 pc(rest);
351 pd(rest-pos.row(4)); // CB->rest bond vector
352 }
353 return n_bead;
354}
