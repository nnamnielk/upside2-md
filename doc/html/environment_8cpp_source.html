498
499 WeightedPos(hid_t grp, CoordNode& pos_, CoordNode& energy_):
500 CoordNode(get_dset_size(1,grp,"index_pos")[0], 4),
501 pos(pos_),
502 energy(energy_),
503 params(n_elem)
504 {
505 traverse_dset<1,int>(grp,"index_pos" ,[&](size_t ne, int x){params[ne].index_pos =x;});
506 traverse_dset<1,int>(grp,"index_weight",[&](size_t ne, int x){params[ne].index_weight=x;});
507 }
508
509 virtual void compute_value(ComputeMode mode) override {
510 Timer timer("weighted_pos");
511
512 for(int ne=0; ne<n_elem; ++ne) {
513 auto p = params[ne];
514 output(0,ne) = pos.output(0,p.index_pos);
515 output(1,ne) = pos.output(1,p.index_pos);
516 output(2,ne) = pos.output(2,p.index_pos);
517 output(3,ne) = expf(-energy.output(0,p.index_weight));
518 }
519 }
520
521 virtual void propagate_deriv() override {
522 Timer timer("d_weighted_pos");
523
524 for(int ne=0; ne<n_elem; ++ne) {
525 auto p = params[ne];
526 pos.sens(0,p.index_pos) += sens(0,ne);
527 pos.sens(1,p.index_pos) += sens(1,ne);
528 pos.sens(2,p.index_pos) += sens(2,ne);
529 energy.sens(0,p.index_weight) -= output(3,ne)*sens(3,ne); // exponential derivative
530 }
531 }

558 virtual void compute_value(ComputeMode mode) override {
559 Timer timer("uniform_transform");
560 for(int ne=0; ne<n_elem; ++ne) {
561 auto coord = (input.output(0,ne)-spline_offset)*spline_inv_dx;
562 auto v = clamped_deBoor_value_and_deriv(bspline_coeff.get(), coord, n_coeff);
563 output(0,ne) = v[0];
564 jac[ne] = v[1]*spline_inv_dx;
565 }
566 }
567
568 virtual void propagate_deriv() override {
569 Timer timer("d_uniform_transform");
570 for(int ne=0; ne<n_elem; ++ne)
571 input.sens(0,ne) += jac[ne]*sens(0,ne);
572 }
573
574 virtual std::vector<float> get_param() const override{
575 vector<float> ret(2+n_coeff);
576 ret[0] = spline_offset;
577 ret[1] = spline_inv_dx;
578 copy_n(bspline_coeff.get(), n_coeff, &ret[2]);
579 return ret;
580 }
581
582#ifdef PARAM_DERIV
583 virtual std::vector<float> get_param_deriv() override {
584 vector<float> ret(2+n_coeff, 0.f);
585 int starting_bin;
586 float d[4];
587 for(int ne=0; ne<n_elem; ++ne) {
588 auto coord = (input.output(0,ne)-spline_offset)*spline_inv_dx;
589 auto v = clamped_deBoor_value_and_deriv(bspline_coeff.get(), coord, n_coeff);
590 clamped_deBoor_coeff_deriv(&starting_bin, d, coord, n_coeff);
591
592 ret[0] += v[1]; // derivative of offset
593 ret[1] += v[1]*(input.output(0,ne)-spline_offset); // derivative of inv_dx
594 for(int i: range(4)) ret[2+starting_bin+i] += d[i];
595 }
596 return ret;
597 }
598#endif
599
600 virtual void set_param(const std::vector<float>& new_param) override {
601 if(new_param.size() < size_t(2+4)) throw string("too small of size for spline");
602 if(int(new_param.size())-2 != n_coeff) {
603 n_coeff = int(new_param.size())-2;
604 bspline_coeff = new_aligned<float>(n_coeff,4);
605 fill_n(bspline_coeff.get(), round_up(n_coeff,4), 0.f);
606 }
607 spline_offset = new_param[0];
608 spline_inv_dx = new_param[1];
609 copy_n(begin(new_param)+2, n_coeff, bspline_coeff.get());
610 }

627 LinearCoupling(hid_t grp, CoordNode* input_, CoordNode* inactivation_):

660 Timer timer("linear_coupling");
661 int n_elem = input.n_elem;
662 float pot = 0.f;
663 for(int ne=0; ne<n_elem; ++ne) {
664 float c = couplings[coupling_types[ne]];
665 float act = inactivation ? sqr(1.f-inactivation->output(inactivation_dim,ne)) : 1.f;
666 float val = input.output(0,ne);
667 pot += c * val * act;
668 input.sens(0,ne) += c*act;
669 if(inactivation) inactivation->sens(inactivation_dim,ne) -= c*val;
670 }
671 potential = pot;
672 }

675 return couplings;
676 }

683 for(int ne=0; ne<n_elem; ++ne) {
684 float act = inactivation ? 1.f - inactivation->output(inactivation_dim,ne) : 1.f;
685 deriv[coupling_types[ne]] += input.output(0,ne) * act;
686 }
687 return deriv;
688 }
689#endif

692 if(new_param.size() != couplings.size())

694 copy(begin(new_param),end(new_param), begin(couplings));
