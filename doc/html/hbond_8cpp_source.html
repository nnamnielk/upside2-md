65 virtual void compute_value(ComputeMode mode) override {
66 Timer timer(string("infer_H_O"));
67
68 VecArray posc = pos.output;
69 for(int nv=0; nv<n_virtual; ++nv) {
70 // For output, first three components are position of H/O and second three are HN/OC bond direction
71 // Bond direction is a unit vector
72 // curr is the N or C atom
73 // This algorithm assumes perfect 120 degree bond angles
74
75 auto& p = params[nv];
76
77 auto prev_c = Float4(&posc(0,p.atom[0]));
78 auto curr_c = Float4(&posc(0,p.atom[1]));
79 auto next_c = Float4(&posc(0,p.atom[2]));
80
81 auto prev = prev_c - curr_c; auto prev_invmag = inv_mag(prev); prev *= prev_invmag;
82 auto next = next_c - curr_c; auto next_invmag = inv_mag(next); next *= next_invmag;
83 auto disp = prev + next ; auto disp_invmag = inv_mag(disp); disp *= disp_invmag;
84
85 auto hbond_dir = -disp;
86 auto hbond_pos = fmadd(Float4(p.bond_length),hbond_dir, curr_c);
87
88 // store derived values for derivatives later
89 prev.blend<0,0,0,1>(prev_invmag).store(data_for_deriv + nv*3*4 + 0);
90 next.blend<0,0,0,1>(next_invmag).store(data_for_deriv + nv*3*4 + 4);
91 disp.blend<0,0,0,1>(disp_invmag).store(data_for_deriv + nv*3*4 + 8);
92
93 // write pos
94 hbond_pos.store(&output(0,nv));
95 hbond_dir.store(&output(3,nv), Alignment::unaligned);
96 }
97 }
98
99 virtual void propagate_deriv() override {
100 Timer timer(string("infer_H_O_deriv"));
101 VecArray pos_sens = pos.sens;
102
103 for(int nv=0; nv<n_virtual; ++nv) {
104 const auto& p = params[nv];
105
106 auto sens_pos = Float4(&sens(0,nv)).zero_entries<0,0,0,1>(); // last entry should be zero
107 auto sens_dir = Float4(&sens(3,nv), Alignment::unaligned);
108
109 auto sens_neg_unitdisp = sens_dir + Float4(p.bond_length)*sens_pos;
110
111 // loading: first 3 entries are unitvec and last is inv_mag
112 auto prev4 = Float4(data_for_deriv + nv*3*4 + 0); auto prev_invmag = prev4.broadcast<3>();
113 auto next4 = Float4(data_for_deriv + nv*3*4 + 4); auto next_invmag = next4.broadcast<3>();
114 auto disp4 = Float4(data_for_deriv + nv*3*4 + 8); auto disp_invmag = disp4.broadcast<3>();
115
116 // use dot3 here so we don't have to zero the last component of disp4, etc
117 auto sens_nonunit_disp = disp_invmag *fmsub(dot3(disp4,sens_neg_unitdisp),disp4, sens_neg_unitdisp);
118 auto sens_nonunit_prev = (-prev_invmag)*fmsub(dot3(prev4,sens_nonunit_disp),prev4, sens_nonunit_disp);
119 auto sens_nonunit_next = (-next_invmag)*fmsub(dot3(next4,sens_nonunit_disp),next4, sens_nonunit_disp);
120
121 sens_nonunit_prev .update(&pos_sens(0,p.atom[0]));
122 (sens_pos - sens_nonunit_prev - sens_nonunit_next).update(&pos_sens(0,p.atom[1]));
123 sens_nonunit_next .update(&pos_sens(0,p.atom[2]));
124 }
125 }

134template <typename S>
135Vec<2,S> hbond_radial_potential(const S& input,
136 const S& inner_barrier, const S& inv_inner_width,
137 const S& outer_barrier, const S& inv_outer_width
138 )
139{
140 Vec<2,S> outer_sigmoid = sigmoid((outer_barrier-input)*inv_outer_width);
141 Vec<2,S> inner_sigmoid = sigmoid((input-inner_barrier)*inv_inner_width);
142
143 return make_vec2( outer_sigmoid.x() * inner_sigmoid.x(),
144 - inv_outer_width * outer_sigmoid.y() * inner_sigmoid.x()
145 + inv_inner_width * inner_sigmoid.y() * outer_sigmoid.x());
146}

149template <typename S>
150Vec<2,S> hbond_angular_potential(const S& dotp, const S& wall_dp, const S& inv_dp_width)
151{
152 Vec<2,S> v = sigmoid((dotp-wall_dp)*inv_dp_width);
153 return make_vec2(v.x(), inv_dp_width*v.y());
154}

174 static Float4 compute_edge(Vec<n_dim1,Float4> &d1, Vec<n_dim2,Float4> &d2, const float* p[4],
175 const Vec<n_dim1,Float4> &x1, const Vec<n_dim2,Float4> &x2) {
176 auto one = Float4(1.f);
177
178 auto H = extract<0,3>(x1);
179 auto O = extract<0,3>(x2);
180 auto rHN = extract<3,6>(x1);
181 auto rOC = extract<3,6>(x2);
182
183 auto HO = H-O;
184
185 auto magHO2 = mag2(HO) + Float4(1e-6f); // a bit of paranoia to avoid division by zero later
186 auto invHOmag = rsqrt(magHO2);
187 auto magHO = magHO2 * invHOmag; // avoid a sqrtf later
188
189 auto rHO = HO*invHOmag;
190
191 auto dotHOC = dot(rHO,rOC);
192 auto dotOHN = -dot(rHO,rHN);
193
194 Vec<3,Float4> dH,dO,drHN,drOC;
195 Float4 hb;
196 auto within_angular_cutoff = (Float4(angular_cutoff) < dotHOC) & (Float4(angular_cutoff) < dotOHN);
197 if(none(within_angular_cutoff)) {
198 dH=dO=drHN=drOC=make_zero<3,Float4>();
199 hb = zero<Float4>();
200 } else {
201 // FIXME I have to load up 4 of these rather pointlessly since they will all be the same
202 // FIXME I don't know if I guarantee alignment on parameters
203 // I expand to 8 to be sure
204 auto p0 = Float4(p[0]);
205 auto p1 = Float4(p[1]);
206 auto p2 = Float4(p[2]);
207 auto p3 = Float4(p[3]); transpose4(p0,p1,p2,p3);
208 auto p4 = Float4(p[0]+4);
209 auto p5 = Float4(p[1]+4);
210 auto p6 = Float4(p[2]+4);
211 auto p7 = Float4(p[3]+4); transpose4(p4,p5,p6,p7);
212
213 auto radial = hbond_radial_potential (magHO , p0, p1, p2, p3); // x has val, y has deriv
214 auto angular1 = hbond_angular_potential(dotHOC, p4, p5);
215 auto angular2 = hbond_angular_potential(dotOHN, p4, p5);
216
217 hb = radial.x() * angular1.x() * angular2.x();
218 auto c0 = radial.y() * angular1.x() * angular2.x();
219 auto c1 = radial.x() * angular1.y() * angular2.x();
220 auto c2 = -radial.x() * angular1.x() * angular2.y();
221
222 drOC = c1*rHO;
223 drHN = c2*rHO;
224
225 dH = c0*rHO + (c1*invHOmag)*(rOC-dotHOC*rHO) + (c2*invHOmag)*(rHN+dotOHN*rHO);
226 dO = -dH;
227 }
228
229 auto hb_log = ternary(one<=hb, Float4(100.f), -logf(one-hb)); // work in multiplicative space
230
231 auto deriv_prefactor = min(rcp(one-hb),Float4(1e5f)); // FIXME this is a mess
232 store<0,3>(d1, dH * deriv_prefactor);
233 store<3,6>(d1, drHN * deriv_prefactor);
234 store<0,3>(d2, dO * deriv_prefactor);
235 store<3,6>(d2, drOC * deriv_prefactor);
236
237 return hb_log;
238 }

271 Float4 one(1.f);
272
273 // print_vector("hb_pos[0]", hb_pos[0]);

363
364 ProteinHBond(hid_t grp, CoordNode& infer_):
365 CoordNode(get_dset_size(1,grp,"index1")[0]+get_dset_size(1,grp,"index2")[0], 7),
366 infer(infer_),
367 igraph(grp, &infer, &infer) ,
368 n_donor (igraph.n_elem1),
369 n_acceptor(igraph.n_elem2),
370 n_virtual (n_donor+n_acceptor),
371 sens_scaled(new_aligned<float>(n_virtual,4))
372 {
373 if(logging(LOG_DETAILED)) {
374 default_logger->add_logger<float>("hbond", {n_donor+n_acceptor}, [&](float* buffer) {
375 for(int nv: range(n_donor+n_acceptor))
376 buffer[nv] = output(6,nv);});
377 }
378 }
379
380 virtual void compute_value(ComputeMode mode) override {
381 Timer timer(string("protein_hbond"));
382
383 int n_virtual = n_donor + n_acceptor;
384 VecArray vs = output;
385 VecArray ho = infer.output;
386
387 for(int nv: range(n_virtual)) {
388 Float4(&ho(0,nv)).store(&vs(0,nv));
389 Float4(&ho(4,nv)).store(&vs(4,nv)); // result is already zero padded
390 }
391
392 // Compute protein hbonding score and its derivative
393 igraph.compute_edges();
394 for(int ne=0; ne<igraph.n_edge; ++ne) {
395 int nd = igraph.edge_indices1[ne];
396 int na = igraph.edge_indices2[ne];
397 float hb_log = igraph.edge_value[ne];
398 vs(6,nd) += hb_log;
399 vs(6,na+n_donor) += hb_log;
400 }
401
402 for(int nv: range(n_virtual)) vs(6,nv) = 1.f-expf(-vs(6,nv));
403 }
404
405 virtual void propagate_deriv() override {
406 Timer timer(string("protein_hbond_deriv"));
407
408 // we accumulated derivatives for z = 1-exp(-log(no_hb))
409 // so we need to convert back with z_sens*(1.f-hb)
410 for(int nv=0; nv<n_virtual; ++nv)
411 sens_scaled[nv] = sens(6,nv) * (1.f-output(6,nv));
412
413 // Push protein HBond derivatives
414 for(int ne: range(igraph.n_edge)) {
415 auto don_sens = sens_scaled[igraph.edge_indices1[ne]];
416 auto acc_sens = sens_scaled[igraph.edge_indices2[ne]+n_donor];
417
418 igraph.edge_sensitivity[ne] = don_sens + acc_sens;
419 }
420 igraph.propagate_derivatives();
421
422 // pass through derivatives on all other components
423 VecArray pd1 = igraph.pos_node1->sens;
424 for(int nd=0; nd<n_donor; ++nd) {
425 // the last component is taken care of by the edge loop
426 update_vec(pd1, igraph.loc1[nd], load_vec<6>(sens, nd));
427 }
428 VecArray pd2 = igraph.pos_node2->sens;
429 for(int na=0; na<n_acceptor; ++na) { // acceptor loop
430 // the last component is taken care of by the edge loop
431 update_vec(pd2, igraph.loc2[na], load_vec<6>(sens, na+n_donor));
432 }
433 }
434};

557 Timer timer(string("hbond_energy"));
558 VecArray pp = protein_hbond.output;
559 VecArray pp_sens = protein_hbond.sens;
560 VecArray ramac = rama.output;
561 VecArray rama_sens = rama.sens;
562
563 for( int i=0; i<n_res; i++) {
564 auto r = load_vec<2>(ramac, i);
565
566 auto phiv1 = compact_sigmoid(r.v[0]-boundary_turn1, sharpness_turn1);
567 auto phiv2 = compact_sigmoid(r.v[0]-boundary_turn2, sharpness_turn2);
568 auto psiv1 = compact_sigmoid(r.v[1]-boundary_helix1, sharpness_helix1);
569 auto psiv2 = compact_sigmoid(r.v[1]-boundary_helix2, sharpness_helix2);
570
571 float turn_score = (1.-phiv1.x())*phiv2.x();
572 float psi_score = (1.-psiv1.x())*psiv2.x();
573 float n_turn_score = 1. - turn_score;
574 float n_psi_score = 1. - psi_score;
575 float helix_score = n_turn_score * psi_score;
576 float sheet_score = n_turn_score * n_psi_score;
577
578 float dturn_dphi = phiv2.y() - (phiv1.y()*phiv2.x() + phiv1.x()*phiv2.y());
579 float dpsis_dpsi = psiv2.y() - (psiv1.y()*psiv2.x() + psiv1.x()*psiv2.y());
580 float dhelix_dphi = -dturn_dphi * psi_score;
581 float dsheet_dphi = -dturn_dphi * n_psi_score;
582 float dhelix_dpsi = n_turn_score * dpsis_dpsi;
583 float dsheet_dpsi = n_turn_score * -dpsis_dpsi;
584
585 Ehbond[i] = E_alpha * helix_score + E_beta * sheet_score + E_other * turn_score;
586 dPhi[i] = E_alpha * dhelix_dphi + E_beta * dsheet_dphi + E_other * dturn_dphi;
587 dPsi[i] = E_alpha * dhelix_dpsi + E_beta * dsheet_dpsi;
588
589 }
590
591 // count hbond number in one residue
592 for( int i =0; i<n_res; i++ )
593 hb_number1[i] = 0.0;
594 for(int nd=0; nd<n_donor; ++nd) {
595 int resid = donor_resid[nd];
596 hb_number1[resid] += pp(6,nd);
597 }
598 for(int na=0; na<n_acceptor; ++na) {
599 int resid = acceptor_resid[na];
600 hb_number1[resid] += pp(6,na+n_donor);
601 }
602
603 // count hbond number in one peptide
604 for( int i =0; i<n_res-1; i++ )
605 hb_number2[i] = 0.0;
606 for(int nd=0; nd<n_donor; ++nd) {
607 int pepid = donor_resid[nd]-1;
608 hb_number2[pepid] += pp(6,nd);
609 }
610 for(int na=0; na<n_acceptor; ++na) {
611 int pepid = acceptor_resid[na];
612 hb_number2[pepid] += pp(6,na+n_donor);
613 }
614
615 n_hbond = 0.0;
616 potential = 0.0;
617 for( int i =0; i<n_res; i++ ) {
618 if (hb_number1[i] == 0.0)
619 continue;
620 n_hbond += hb_number1[i];
621 potential += hb_number1[i]*Ehbond[i];
622
623 auto iter = resid_donor.find(i);
624 if(iter!=resid_donor.end()) {
625 pp_sens(6,iter->second) += Ehbond[i];
626 }
627 iter = resid_acceptor.find(i);
628 if(iter!=resid_acceptor.end()) {
629 pp_sens(6,iter->second+n_donor) += Ehbond[i];
630 }
631
632 rama_sens(0, i) += hb_number1[i]*dPhi[i];
633 rama_sens(1, i) += hb_number1[i]*dPsi[i];
634 }
635
636 for( int i =0; i<n_res-1; i++ ) {
637 if (hb_number2[i] <= 1.0)
638 continue;
639 auto dE = compact_sigmoid(1.5-hb_number2[i], 2.0);
640 potential += dE.x() * E_bias;
641 auto dE_dHB = -dE.y() * E_bias ;
642 auto iter = resid_donor.find(i+1);
643 if(iter!=resid_donor.end()) {
644 pp_sens(6,iter->second) += dE_dHB;
645 }
646 iter = resid_acceptor.find(i);
647 if(iter!=resid_acceptor.end()) {
648 pp_sens(6,iter->second+n_donor) += dE_dHB;
649 }
650 }
