36// This file provides some simple tools that can be used to convert
37// integers of various widths to floats and doubles with various
38// characteristics. It can be used to generate real-valued, uniformly
39// distributed random variables from the random integers produced by
40// the Random123 CBRNGs.
41//
42// There are three templated functions:
43//
44// u01: output is as dense as possible in (0,1}, never 0.0. May
45// return 1.0 if and only if the number of output mantissa bits
46// is less than the width of the input.
47// uneg11: output is as dense as possible in {-1,1}, never 0.0. May
48// return 1.0 or -1.0 if and only if the number of output mantissa bits
49// is less than the width of the input.
50// u01fixedpt: output is "fixed point", equispaced, open at both ends,
51// and is never 0.0, 0.5 nor 1.0.
52//
53// The behavior of u01 and uneg11 depend on the pre-processor symbol:
54// R123_UNIFORM_FLOAT_STORE. When #defined to a non-zero value, u01
55// and uneg11 declare a volatile intermediate result, with the
56// intention of forcing architectures that have "extra bits" in their
57// floating point registers to more closely conform to IEEE
58// arithmetic. When compiled this way, u01 and uneg11 will be
59// significantly slower, as they will incur a memory write and read on
60// every call. Without it, they may fail the "known answer test"
61// implemented in ut_uniform_IEEEkat.cpp even though they perform
62// perfectly reasonable int to float conversions. We have used
63// this option to get 32-bit x86 to produce the same results as
64// 64-bit x86-64 code, but we do not recommend it for normal
65// use.
66//
67// This file may not be as portable, and has not been tested as
68// rigorously as the files in the library itself, i.e., those in
69// ../include/Random123. Nevertheless, we hope it is useful and we
70// encourage developers to copy it and modify it for their own
71// use. We invite comments and improvements.
72

107#if defined(__CUDACC__) || defined(_LIBCPP_HAS_NO_CONSTEXPR)
108// Amazing! cuda thinks numeric_limits::max() is a __host__ function, so
109// we can't use it in a device function.
110//
111// The LIBCPP_HAS_NO_CONSTEXP test catches situations where the libc++
112// library thinks that the compiler doesn't support constexpr, but we
113// think it does. As a consequence, the library declares
114// numeric_limits::max without constexpr. This workaround should only
115// affect a narrow range of compiler/library pairings.
116//
117// In both cases, we find max() by computing ~(unsigned)0 right-shifted
118// by is_signed.
119template <typename T>
120R123_CONSTEXPR R123_STATIC_INLINE R123_CUDA_DEVICE T maxTvalue(){
121 typedef typename make_unsigned<T>::type uT;
122 return (~uT(0)) >> std::numeric_limits<T>::is_signed;
123 }
124#else
125template <typename T>
126R123_CONSTEXPR R123_STATIC_INLINE T maxTvalue(){
127 return std::numeric_limits<T>::max();
128}
129#endif

131// u01: Input is a W-bit integer (signed or unsigned). It is cast to
132// a W-bit unsigned integer, multiplied by Ftype(2^-W) and added to
133// Ftype(2^(-W-1)). A good compiler should optimize it down to an
134// int-to-float conversion followed by a multiply and an add, which
135// might be fused, dependingon the architecture.
136//
137// If the input is a uniformly distributed integer, then the
138// result is a uniformly distributed floating point number in [0, 1].
139// The result is never exactly 0.0.
140// The smallest value returned is 2^-W.
141// Let M be the number of mantissa bits in Ftype.
142// If W>M then the largest value retured is 1.0.
143// If W<=M then the largest value returned is the largest Ftype less than 1.0.
144template <typename Ftype, typename Itype>
145R123_CUDA_DEVICE R123_STATIC_INLINE Ftype u01(Itype in){
146 typedef typename make_unsigned<Itype>::type Utype;
147 R123_CONSTEXPR Ftype factor = Ftype(1.)/(maxTvalue<Utype>() + Ftype(1.));
148 R123_CONSTEXPR Ftype halffactor = Ftype(0.5)*factor;
149#if R123_UNIFORM_FLOAT_STORE
150 volatile Ftype x = Utype(in)*factor; return x+halffactor;
151#else
152 return Utype(in)*factor + halffactor;
153#endif
154}

156// uneg11: Input is a W-bit integer (signed or unsigned). It is cast
157// to a W-bit signed integer, multiplied by Ftype(2^-(W-1)) and
158// then added to Ftype(2^(-W-2)). A good compiler should optimize
159// it down to an int-to-float conversion followed by a multiply and
160// an add, which might be fused, depending on the architecture.
161//
162// If the input is a uniformly distributed integer, then the
163// output is a uniformly distributed floating point number in [-1, 1].
164// The result is never exactly 0.0.
165// The smallest absolute value returned is 2^-(W-1)
166// Let M be the number of mantissa bits in Ftype.
167// If W>M then the largest value retured is 1.0 and the smallest is -1.0.
168// If W<=M then the largest value returned is the largest Ftype less than 1.0
169// and the smallest value returned is the smallest Ftype greater than -1.0.
170template <typename Ftype, typename Itype>
171R123_CUDA_DEVICE R123_STATIC_INLINE Ftype uneg11(Itype in){
172 typedef typename make_signed<Itype>::type Stype;
173 R123_CONSTEXPR Ftype factor = Ftype(1.)/(maxTvalue<Stype>() + Ftype(1.));
174 R123_CONSTEXPR Ftype halffactor = Ftype(0.5)*factor;
175#if R123_UNIFORM_FLOAT_STORE
176 volatile Ftype x = Stype(in)*factor; return x+halffactor;
177#else
178 return Stype(in)*factor + halffactor;
179#endif
180}

182// u01fixedpt: Return a "fixed point" number in (0,1). Let:
183// W = width of Itype, e.g., 32 or 64, regardless of signedness.
184// M = mantissa bits of Ftype, e.g., 24, 53 or 64
185// B = min(M, W)
186// Then the 2^B-1 possible output values are:
187// 2^-B*{1, 3, 5, ..., 2^B - 1}
188// The smallest output is: 2^-B
189// The largest output is: 1 - 2^-B
190// The output is never exactly 0.0, nor 0.5, nor 1.0.
191// The 2^(B-1) possible outputs:
192// - are equally likely,
193// - are uniformly spaced by 2^-(B-1),
194// - are balanced around 0.5
195template <typename Ftype, typename Itype>
196R123_CUDA_DEVICE R123_STATIC_INLINE Ftype u01fixedpt(Itype in){
197 typedef typename make_unsigned<Itype>::type Utype;
198 R123_CONSTEXPR int excess = std::numeric_limits<Utype>::digits - std::numeric_limits<Ftype>::digits;
199 if(excess>=0){
200 R123_CONSTEXPR int ex_nowarn = (excess>=0) ? excess : 0;
201 R123_CONSTEXPR Ftype factor = Ftype(1.)/(Ftype(1.) + ((maxTvalue<Utype>()>>ex_nowarn)));
202 return (1 | (Utype(in)>>ex_nowarn)) * factor;
203 }else
204 return u01<Ftype>(in);
205}
