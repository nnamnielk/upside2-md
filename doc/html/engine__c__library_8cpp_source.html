210 int i;
211 for(i=0; i<nx-3; i+=4) {

213 en.x().store(result+i, Alignment::unaligned);

216 if(i<nx) {

221 en.x().store(last_data);

238int get_bounded_value_and_deriv_2d(int N, float* result, const float* bspline_coeff, int nx, const float* x) {
239 // this is a rather convoluted function because there is no scalar interface to the 2d splines
240 const float* p[4] = {bspline_coeff, bspline_coeff, bspline_coeff, bspline_coeff};
241
242 for(int i: range(0,nx,4)) {
243 alignas(16) float x4[4] = {0.f,0.f,0.f,0.f};
244 alignas(16) float y4[4] = {0.f,0.f,0.f,0.f};
245 for(int j=0; j<4; ++j) {
246 x4[j] = j<min(4,nx-i) ? x[2*(i+j) ] : 1.f;
247 y4[j] = j<min(4,nx-i) ? x[2*(i+j)+1] : 1.f;
248
249 if((x4[j]<1.f)|(x4[j]>=float(N-2))|(y4[j]<1.f)|(y4[j]>=float(N-2))) return 1;
250 }
251
252 auto v = deBoor2d_value_and_deriv(N, p, Float4(x4), Float4(y4));
253
254 alignas(16) float val[4] = {0.f,0.f,0.f,0.f};
255 alignas(16) float der_x[4] = {0.f,0.f,0.f,0.f};
256 alignas(16) float der_y[4] = {0.f,0.f,0.f,0.f};
257 v[0].store(val);
258 v[1].store(der_x);
259 v[2].store(der_y);
260
261 for(int j=0; j<min(4,nx-i); ++j) {
262 result[(i+j)*3+0] = val[j];
263 result[(i+j)*3+1] = der_x[j];
264 result[(i+j)*3+2] = der_y[j];
265 }
266 }
267 return 0;
268}
