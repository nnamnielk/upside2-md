18template <bool return_deriv>
19inline float
20nonbonded_kernel_or_deriv_over_r(float r_mag2)
21{
22 const float energy_scale = 4.f;
23 const float wall = 3.0f; // corresponds to vdW *diameter*
24 const float wall_squared = wall*wall;
25 const float width = 0.10f;
26 const float sharpness = 1.f/(wall*width); // ensure character
27
28 const float2 V = energy_scale*compact_sigmoid(r_mag2-wall_squared, sharpness);
29 return return_deriv ? 2.f*V.y() : V.x();
