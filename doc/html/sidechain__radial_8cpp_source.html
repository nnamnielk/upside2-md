17struct RadialHelper {
18 // spline-based distance interaction
19 // n_knot is the number of basis splines (including those required to get zero
20 // derivative in the clamped spline)
21 // spline is constant over [0,dx] to avoid funniness at origin
22
23 // Please obey these 4 conditions:
24 // p[0] = 1./dx, that is the inverse of the knot spacing
25 // should have p[1] == p[3] for origin clamp (p[0] is inv_dx)
26 // should have p[-3] == p[-1] (negative indices from the end, Python-style) for terminal clamping
27 // should have (1./6.)*p[-3] + (2./3.)*p[-2] + (1./6.)*p[-1] == 0. for continuity at cutoff
28
29 constexpr static bool symmetric = is_symmetric;
30 constexpr static int n_knot=16, n_param=1+n_knot, n_dim1=3, n_dim2=3, simd_width=1;
31
32 static float cutoff(const float* p) {
33 const float inv_dx = p[0];
34 return (n_knot-2-1e-6)/inv_dx; // 1e-6 just insulates us from round-off error
35 }
36
37 static bool is_compatible(const float* p1, const float* p2) {
38 if(symmetric) for(int i: range(n_param)) if(p1[i]!=p2[i]) return false;
39 return true;
40 }
41
42 static Int4 acceptable_id_pair(const Int4& id1, const Int4& id2) {
43 auto sequence_exclude = Int4(2);
44 return (sequence_exclude < id1-id2) | (sequence_exclude < id2-id1);
45 }
46
47 static Float4 compute_edge(Vec<n_dim1,Float4> &d1, Vec<n_dim2,Float4> &d2, const float* p[4],
48 const Vec<n_dim1,Float4> &x1, const Vec<n_dim2,Float4> &x2) {
49 alignas(16) const float inv_dx_data[4] = {p[0][0], p[1][0], p[2][0], p[3][0]};
50
51 auto inv_dx = Float4(inv_dx_data, Alignment::aligned);
52 auto disp = x1-x2;
53 auto dist2 = mag2(disp);
54 auto inv_dist = rsqrt(dist2+Float4(1e-7f)); // 1e-7 is divergence protection
55 auto dist_coord = dist2*(inv_dist*inv_dx);
56
57 const float* pp[4] = {p[0]+1, p[1]+1, p[2]+1, p[3]+1};
58 auto en = clamped_deBoor_value_and_deriv(pp, dist_coord, n_knot);
59 d1 = disp*(inv_dist*inv_dx*en.y());
60 d2 = -d1;
61 return en.x();
62 }
63
64 static void param_deriv(Vec<n_param> &d_param, const float* p,
65 const Vec<n_dim1> &x1, const Vec<n_dim2> &x2) {
66 d_param = make_zero<n_param>();
67 float inv_dx = p[0];
68 auto dist_coord = inv_dx*mag(x1-x2); // to convert to spline coords of interger grid of knots
69 auto dV_dinv_dx = clamped_deBoor_value_and_deriv(p+1, dist_coord, n_knot).y()*mag(x1-x2);
70 d_param[0] = dV_dinv_dx;
71
72 int starting_bin;
73 float result[4];
74 clamped_deBoor_coeff_deriv(&starting_bin, result, dist_coord, n_knot);
75 for(int i: range(4)) d_param[1+starting_bin+i] = result[i];
76 }
77};

138struct ContactInteraction {
139
140 constexpr static bool symmetric = is_symmetric;
141 constexpr static int n_param = 4, n_dim1=3, n_dim2=3, simd_width=1;
142
143 static float cutoff(const float* p) {
144 return p[3]+1e-6;
145 }
146
147 static Int4 acceptable_id_pair(const Int4& id1, const Int4& id2) {
148 auto sequence_exclude = Int4(2); // exclude i,i, i,i+1, and i,i+2
149 return (sequence_exclude < id1-id2) | (sequence_exclude < id2-id1);
150 }
151
152 static bool is_compatible(const float* p1, const float* p2) { return true;}
153
154 static Float4 compute_edge(Vec<n_dim1,Float4> &d1, Vec<n_dim2,Float4> &d2, const float* p[4],
155 const Vec<n_dim1,Float4> &x1, const Vec<n_dim2,Float4> &x2) {
156
157 auto disp = x1-x2;
158 auto dist2 = mag2(disp);
159 auto inv_dist = rsqrt(dist2+Float4(1e-7f)); // 1e-7 is divergence protection
160 auto dist = dist2*inv_dist;
161 auto scale = Float4(p[0]);
162 auto center = Float4(p[1]);
163 auto sharpness = Float4(p[2]);
164 auto empty = Float4(p[3]);
165 transpose4(scale, center, sharpness, empty);
166 auto sig = compact_sigmoid(dist-center, sharpness);
167 d1 = disp*inv_dist*scale*sig.y();
168 d2 = -d1;
169 return scale*sig.x();
170 }
171
172 static void param_deriv(Vec<n_param> &d_param, const float* p,
173 const Vec<n_dim1> &x1, const Vec<n_dim2> &x2) {
174 d_param = make_zero<n_param>();
175 auto dist = mag(x1-x2);
176 auto out = compact_sigmoid(dist-p[1], p[2]);
177 d_param[0] += out.x();
178 d_param[1] += -p[0] * out.y();
179 d_param[2] += p[0] * (dist-p[1]) * out.y()/p[2];
180 }
