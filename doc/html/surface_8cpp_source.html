25void RotateXY(vector<float>& x, vector<float>& y, float rx, float ry, float angle, int size) {
26 for (int na=0; na< size; ++na ) {
27 float dx = x[na] - rx;
28 float dy = y[na] - ry;
29 if (dx==0 and dy==0) continue;
30 x[na] = rx + cos(angle) * dx - sin(angle) * dy;
31 y[na] = ry + sin(angle) * dx + cos(angle) * dy;
32 }
33}

35void FindMinMax(vector<float>& X, float& minx, float& maxx, bool initialization, int size ) {
36 int ii = 0;
