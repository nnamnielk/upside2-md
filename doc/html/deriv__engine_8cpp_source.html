13void
14integration_stage(
15 VecArray mom,
16 VecArray pos,
17 const VecArray deriv,
18 float vel_factor,
19 float pos_factor,
20 float max_force,
21 int n_atom)
22{
23 for(int na=0; na<n_atom; ++na) {
24 // assumes unit mass for all particles
25
26 auto d = load_vec<3>(deriv, na);
27 if(max_force) {
28 float f_mag = mag(d)+1e-6f; // ensure no NaN when mag(deriv)==0.
29 float scale_factor = atan(f_mag * ((0.5f*M_PI_F) / max_force)) * (max_force/f_mag * (2.f/M_PI_F));
30 d *= scale_factor;
31 }
32
33 auto p = load_vec<3>(mom, na) - vel_factor*d;
34 store_vec (mom, na, p);
35 update_vec(pos, na, pos_factor*p);
36 }
37}

295void DerivEngine::integration_cycle(VecArray mom, float dt, float max_force, IntegratorType type) {
296 // integrator from Predescu et al., 2012
297 // http://dx.doi.org/10.1080/00268976.2012.681311
298
299 float a = (type==Predescu) ? 0.108991425403425322 : 1./6.;
300 float b = (type==Predescu) ? 0.290485609075128726 : 1./3.;
301
302 float mom_update[] = {1.5f-3.f*a, 1.5f-3.f*a, 6.f*a};
303 float pos_update[] = { 3.f*b, 3.0f-6.f*b, 3.f*b};
304
305 for(int stage=0; stage<3; ++stage) {
306 compute(DerivMode); // compute derivatives
307 Timer timer(string("integration"));
308 integration_stage(
309 mom,
310 pos->output,
311 pos->sens,
312 dt*mom_update[stage], dt*pos_update[stage], max_force,
313 pos->n_atom);
314 }
315}

353DerivEngine initialize_engine_from_hdf5(int n_atom, hid_t potential_group)
354{
355 DerivEngine engine(n_atom);
356 auto& m = node_creation_map();
357
358 map<string, pair<bool,vector<string>>> dep_graph; // bool indicates node is active
359 dep_graph["pos"] = make_pair(true, vector<string>());
360 for(const auto &name : node_names_in_group(potential_group, "."))
361 dep_graph[name] = make_pair(true, read_attribute<vector<string>>(potential_group, name.c_str(), "arguments"));
362
363 for(auto &kv : dep_graph) {
364 for(auto& dep_name : kv.second.second) {
365 if(dep_graph.find(dep_name) == end(dep_graph))
366 throw string("Node ") + kv.first + " takes " + dep_name +
367 " as an argument, but no node of that name can be found.";
368 }
369 }
370
371 vector<string> topo_order;
372 auto in_topo = [&](const string &name) {
373 return find(begin(topo_order), end(topo_order), name) != end(topo_order);};
374
375 int graph_size = dep_graph.size();
376 for(int round_num=0; round_num<graph_size; ++round_num) {
377 for(auto it=begin(dep_graph); it!=end(dep_graph); ++it) {
378 if(!it->second.first) continue;
379 if(all_of(begin(it->second.second), end(it->second.second), in_topo)) {
380 topo_order.push_back(it->first);
381 it->second.first = false; // make node inactive
382 }
383 }
384 }
385 for(auto &kv : dep_graph) if(kv.second.first)
386 throw string("Unsatisfiable dependency ") + kv.first + " in potential computation";
387
388 // using topo_order here ensures that a node is only parsed after all its arguments
389 for(auto &nm : topo_order) {
390 // if(!quiet) printf("initializing %-27s%s", nm.c_str(), nm=="pos" ? "\n" : "");
391 if(nm=="pos") continue; // pos node is added specially
392 // some name in the node_creation_map must be a prefix of this name
393 string node_type_name = "";
394 for(auto &kv : m) {
395 if(is_prefix(kv.first, nm))
396 node_type_name = kv.first;
397 }
398 if(node_type_name == "") throw string("No node type found for name '") + nm + "'";
399 NodeCreationFunction& node_func = m[node_type_name];
400
401 auto argument_names = read_attribute<vector<string>>(potential_group, nm.c_str(), "arguments");
402 int integrator_level = read_attribute<int>(potential_group, nm.c_str(), "integrator_level", 2);
403
404 ArgList arguments;
405
406 for(const auto& arg_name : argument_names) {
407 // if the node is not a CoordNode, a null pointer will be returned from dynamic_cast
408 arguments.push_back(dynamic_cast<CoordNode*>(engine.get(arg_name).computation.get()));
409 if(!arguments.back())
410 throw arg_name + " is not an intermediate value, but it is an argument of " + nm;
411 }
412
413 try {
414 auto grp = open_group(potential_group,nm.c_str());
415 auto g_get = grp.get();
416 auto computation = unique_ptr<DerivComputation>(node_func(g_get, arguments));
417 engine.add_node(nm, integrator_level, move(computation), argument_names);
418 } catch(const string &e) {
419 throw "while adding '" + nm + "', " + e;
420 }
421 }
422
423 engine.build_exec_levels();
424
425 return engine;
426}

447vector<float> central_difference_deriviative(
448 const function<void()>& compute_value, vector<float>& input, vector<float>& output, float eps,
449 ValueType value_type)
450{
451 // FIXME only handles single systems
452 auto old_input = input;
453 compute_value();
454 auto output_minus_eps = output;
455
456 vector<float> jacobian(input.size()*output.size(), -10101.f);
457
458 for(unsigned ni=0; ni<input.size(); ++ni) {
459 copy(begin(old_input), end(old_input), begin(input));
460
461 input[ni] = old_input[ni] - eps;
462 compute_value();
463 copy(begin(output), end(output), begin(output_minus_eps));
464
465 input[ni] = old_input[ni] + eps;
466 compute_value();
467
468 if(value_type == BODY_VALUE) {
469 if(output.size() % 7) throw "impossible";
470 for(unsigned no=0; no<output.size(); no+=7){
471 const float* o = &output[no+3];
472 float* ome = &output_minus_eps[no+3];
473
474 float4 qo = make_vec4(o [0],o [1],o [2],o [3]);
475 float4 qome = make_vec4(ome[0],ome[1],ome[2],ome[3]);
476
477 // resolve whether q or -q is closer
478 if(mag2(qo+qome) < mag2(qo-qome)) for(int d=0; d<4; ++d) ome[d] *= -1.f;
479 }
480 }
481
482 for(unsigned no=0; no<output.size(); ++no){
483 float diff = output[no]-output_minus_eps[no];
484 if(value_type == ANGULAR_VALUE) {
485 //printf("diff %f\n", diff/M_PI_F*180.f);
486 if(diff> M_PI_F) diff -= 2.f*M_PI_F;
487 if(diff<-M_PI_F) diff += 2.f*M_PI_F;
488 }
489 jacobian[no*input.size() + ni] = diff * (0.5f/eps);
490 }
491 }
492
493 // restore input and recompute the output so the caller is not surprised
494 copy(begin(old_input), end(old_input), begin(input));
495 compute_value();
496
497 return jacobian;
498}
