145 for(int aa: range(20))
146 env_cov_sens(0, p.env_index*20+aa) += weights[aa] * d_pot;

153

293 Timer timer(string("cb_membrane_potential"));
294
295 VecArray cb_pos = res_pos.output;
296 VecArray cb_pos_sens = res_pos.sens;
297 VecArray env_cov = environment_coverage.output;
298 VecArray env_cov_sens = environment_coverage.sens;
299
300 potential = 0.f;
301
302 vector<float> bl_score1(n_bl-1);
303 vector<float> bl_score2(n_bl-1);
304 vector<float> d_bl_score1(n_bl-1);
305 vector<float> d_bl_score2(n_bl-1);
306 vector<float> bl_score(n_bl);
307 vector<float> d_bl_score(n_bl);
308
309 vector<float> fs(n_bl);
310 vector<float> dfs(n_bl);
311
312 auto ccenter = load_vec<3>(center_of_curvature.output, 0);
313 curvature_radius = ccenter.z()*-1.f*curvature_sign;
314
315 for(int nr=0; nr<n_elem; ++nr) {
316 int rt = res_type[nr];
317 int ri = cb_index[nr];
318
319 // burial level
320 float bl = 0.0;
321 for(int aa: range(n_restype))
322 bl += weights[aa] * env_cov(0, ri*n_restype+aa);
323
324 for (int i: range(n_bl-1)) {
325 auto s1 = compact_sigmoid(bl-bl_nodes[i], bl_range_sharpness);
326 auto s2 = compact_sigmoid(bl_nodes[i]-bl, bl_range_sharpness);
327 bl_score1[i] = s1.x();
328 d_bl_score1[i] = s1.y();
329 bl_score2[i] = s2.x();
330 d_bl_score2[i] = -s2.y();
331 }
332
333 bl_score[0] = bl_score1[0];
334 d_bl_score[0] = d_bl_score1[0];
335 bl_score[n_bl-1] = bl_score2[n_bl-2];
336 d_bl_score[n_bl-1] = d_bl_score2[n_bl-2];
337 for (int i: range(n_bl-2)) {
338 bl_score[i+1] = bl_score2[i]*bl_score1[i+1];
339 d_bl_score[i+1] = d_bl_score2[i]*bl_score1[i+1] + bl_score2[i]*d_bl_score1[i+1];
340 }
341
342 // z
343 float cb_z = cb_pos(2, ri);
344 float dist = 0.f;
345 float r_dist = 1.f;
346 auto xyz = load_vec<3>(cb_pos, ri);
347
348 if (use_curvature) {
349 dist = mag( xyz - ccenter );
350 r_dist = 1.f/dist;
351 cb_z = curvature_sign* ( dist - curvature_radius );
352 }
353
354 auto sig_left = compact_sigmoid(cb_z-left_right_node, left_right_sharpness);
355 float ll = sig_left.x();
356 float lr = 1.f-sig_left.x();
357 float dll = sig_left.y();
358 float dlr = -sig_left.y();
359
360 auto coord_left = ( cb_z-zstart)*zscale;
361 auto coord_right = (-cb_z-zstart)*zscale;
362
363 float df_dx = 0.0;
364 float df_dy = 0.0;
365 float df_dz = 0.0;
366 float df_dbl = 0.0;
367 for (int i: range(n_bl)) {
368 int shift = (rt*n_bl + i)*n_node;
369 auto v = clamped_deBoor_value_and_deriv(coeff.data() + shift, coord_left, n_node);
370 float f_left = v.x();
371 float df_left = v.y()*zscale;
372
373 v = clamped_deBoor_value_and_deriv(coeff.data() + shift, coord_right, n_node);
374 float f_right = v.x();
375 float df_right = -v.y()*zscale;
376
377 float fs = ll*f_left + lr*f_right;
378 float dfs = ll*df_left + dll*f_left + lr*df_right + dlr*f_right;
379
380 potential += bl_score[i] * fs;
381 df_dbl += d_bl_score[i] * fs;
382 if (use_curvature) {
383 float sen_d = curvature_sign * bl_score[i] * dfs * r_dist;
384 df_dx += sen_d * xyz.x();
385 df_dy += sen_d * xyz.y();
386 df_dz += sen_d * (xyz.z()-curvature_sign*curvature_radius*-1.f);
387 }
388 else
389 df_dz += bl_score[i] * dfs;
390 }
391
392 if (use_curvature) {
393 cb_pos_sens (0, ri) += df_dx;
394 cb_pos_sens (1, ri) += df_dy;
395 }
396 cb_pos_sens (2, ri) += df_dz;
397 for(int aa: range(n_restype))
398 env_cov_sens(0, ri*n_restype+aa) += weights[aa] * df_dbl;
399 }

558 dist = mag( xyz - ccenter );
559 r_dist = 1.f/dist;
560 hb_z = curvature_sign* ( dist - curvature_radius );
561 }

564 auto coord_right = (-hb_z-zstart)*zscale;

569 float dll = sig_left.y();
570 float dlr = -sig_left.y();

574 float fs1 = ll*f1l.x() + lr*f1r.x();

576

578 float sqr_uhb_prob = sqr(uhb_prob);
579 potential += sqr_uhb_prob * fs1;
580 hb_sens(6, ri) += -2.f*uhb_prob * fs1;
581
582
583 auto f2l = clamped_deBoor_value_and_deriv(coeff.data() + (rt*2+1)*n_node, coord_left, n_node);
584 auto f2r = clamped_deBoor_value_and_deriv(coeff.data() + (rt*2+1)*n_node, coord_right, n_node);
585 float fs2 = ll*f2l.x() + lr*f2r.x();
586 float dfs2 = ll*f2l.y()*zscale + dll*f2l.x() - lr*f2r.y()*zscale + dlr*f2r.x();
587
588 float sqr_hb_prob = 1.f - sqr(1.f-hb_prob);
589 potential += sqr_hb_prob * fs2;
590 hb_sens(6, ri) += 2.f*uhb_prob * fs2;
591
592 if (use_curvature) {
593 float sen_d = curvature_sign * sqr_uhb_prob * dfs1 * r_dist;
594 hb_sens(0, ri) += sen_d * xyz.x();
595 hb_sens(1, ri) += sen_d * xyz.y();
596 hb_sens(2, ri) += sen_d * (xyz.z()-curvature_sign*curvature_radius*-1.f);
597 sen_d = curvature_sign * sqr_hb_prob * dfs2 * r_dist;
598 hb_sens(0, ri) += sen_d * xyz.x();
599 hb_sens(1, ri) += sen_d * xyz.y();
600 hb_sens(2, ri) += sen_d * (xyz.z()-curvature_sign*curvature_radius*-1.f);
601 }
602 else {
603 hb_sens(2, ri) += sqr_uhb_prob * dfs1;
604 hb_sens(2, ri) += sqr_hb_prob * dfs2;
605 }
606 }
607 }

838 surfv = 1.0;
839 else
840 surfv = surf(0, ri);

845 float df_left = v.y() * zscale * surfv;
846
847 v = clamped_deBoor_value_and_deriv(coeff.data() + shift, coord_right, n_node);
848 float f_right = v.x() * surfv;
849 float df_right = -v.y() * surfv * zscale;
850
851 if ( surfv < 1.f) {
852 int shift = rt*n_node;
853 auto v = clamped_deBoor_value_and_deriv(coeff_inner.data() + shift, coord_left, n_node);
854 f_left += v.x() * (1.f-surfv);
855 df_left += v.y() * (1.f-surfv) * zscale;
856
857 v = clamped_deBoor_value_and_deriv(coeff_inner.data() + shift, coord_right, n_node);
858 f_right += v.x() * (1.f-surfv);
859 df_right -= v.y() * (1.f-surfv) * zscale;
860 }
861
862 float fs = ll*f_left + lr*f_right;
863 float dfs = ll*df_left + dll*f_left + lr*df_right + dlr*f_right;
864
865 potential += bl_score[i] * fs;

870 df_dy += sen_d * xyz.y();
871 df_dz += sen_d * (xyz.z()-curvature_sign*curvature_radius*-1.f);
872 }
873 else
874 df_dz += bl_score[i] * dfs;
875 }
876
877 if (use_curvature) {

879 cb_pos_sens (1, ri) += df_dy;
880 }
881 cb_pos_sens (2, ri) += df_dz;
882 for(int aa: range(n_restype))
883 env_cov_sens(0, ri*n_restype+aa) += weights[aa] * df_dbl;
884 }
