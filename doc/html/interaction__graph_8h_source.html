51 template<acceptable_id_pair_t acceptable_id_pair>
52 void ensure_cache_valid(
53 float cutoff,
54 const float* aligned_pos1, const int pos1_stride, int* id1,
55 const float* aligned_pos2, const int pos2_stride, int* id2)
56 {
57 Timer t1("pairlist_cache_check");
58 // Find maximum deviation from cached positions to determine if cache must be rebuilt
59 auto max_dist_exceeded = Float4();
60 auto id_changed = Int4();
61 auto max_cache_dist2 = Float4(sqr(0.5f*(cache_cutoff - cutoff)));
62
63 for(int i=0; i<n_elem1; i+=4) {
64 auto x = Float4(aligned_pos1+pos1_stride*(i+0)) - Float4(cache_pos1+4*(i+0));
65 auto y = Float4(aligned_pos1+pos1_stride*(i+1)) - Float4(cache_pos1+4*(i+1));
66 auto z = Float4(aligned_pos1+pos1_stride*(i+2)) - Float4(cache_pos1+4*(i+2));
67 auto w = Float4(aligned_pos1+pos1_stride*(i+3)) - Float4(cache_pos1+4*(i+3));
68
69 transpose4(x,y,z,w);
70 max_dist_exceeded |= max_cache_dist2 < x*x+y*y+z*z;
71
72 // To ensure the caching is completely transparent, we must also ensure that the id's have not
73 // changed. Hopefully, this check is quite quick.
74 id_changed |= Int4(id1+i)!=Int4(cache_id1+i);
75 }
76 if(!symmetric) {
77 for(int i=0; i<n_elem2; i+=4) {
78 auto x = Float4(aligned_pos2+pos2_stride*(i+0)) - Float4(cache_pos2+4*(i+0));
79 auto y = Float4(aligned_pos2+pos2_stride*(i+1)) - Float4(cache_pos2+4*(i+1));
80 auto z = Float4(aligned_pos2+pos2_stride*(i+2)) - Float4(cache_pos2+4*(i+2));
81 auto w = Float4(aligned_pos2+pos2_stride*(i+3)) - Float4(cache_pos2+4*(i+3));
82
83 transpose4(x,y,z,w);
84 max_dist_exceeded |= max_cache_dist2 < x*x+y*y+z*z;
85 id_changed |= Int4(id2+i)!=Int4(cache_id2+i);
86 }
87 }
88 t1.stop();
89
90 // We don't do early bailout since the cache should be valid most of the time
91 if(cache_valid && max_dist_exceeded.none() && id_changed.none()) return;
92 // printf("cache rebuild\n");
93
94 // If we reach here, we must rebuild the cache
95
96 Timer t2("pairlist_cache_rebuild");
97 // Store the new cache positions
98 cache_cutoff = cutoff + cache_buffer;
99
100 for(int i=0; i<n_elem1; i+=4) {
101 Float4(aligned_pos1+pos1_stride*(i+0)).store(cache_pos1+4*(i+0));
102 Float4(aligned_pos1+pos1_stride*(i+1)).store(cache_pos1+4*(i+1));
103 Float4(aligned_pos1+pos1_stride*(i+2)).store(cache_pos1+4*(i+2));
104 Float4(aligned_pos1+pos1_stride*(i+3)).store(cache_pos1+4*(i+3));
105 Int4(id1+i).store(cache_id1+i);
106 }
107 if(!symmetric) {
108 for(int i=0; i<n_elem2; i+=4) {
109 Float4(aligned_pos2+pos2_stride*(i+0)).store(cache_pos2+4*(i+0));
110 Float4(aligned_pos2+pos2_stride*(i+1)).store(cache_pos2+4*(i+1));
111 Float4(aligned_pos2+pos2_stride*(i+2)).store(cache_pos2+4*(i+2));
112 Float4(aligned_pos2+pos2_stride*(i+3)).store(cache_pos2+4*(i+3));
113 Int4(id2+i).store(cache_id2+i);
114 }
115 }
116
117 // Find all cache pairs
118 alignas(16) int32_t offset_v[4] = {0,1,2,3};
119 Int4 offset(offset_v);
120 auto cutoff2 = Float4(sqr(cache_cutoff));
121
122 int ne = 0;
123 for(int32_t i1=0; i1<n_elem1; i1+=4) {
124 Float4 v0(cache_pos1+(i1+0)*4), // aligned_pos1 size was rounded up
125 v1(cache_pos1+(i1+1)*4),
126 v2(cache_pos1+(i1+2)*4),
127 v3(cache_pos1+(i1+3)*4);
128 transpose4(v0,v1,v2,v3); // v3 will be unused at the end
129 auto x1 = make_vec3(v0,v1,v2);
130
131 auto my_id1 = Int4(cache_id1+i1);
132 auto i1_vec = Int4(i1) + offset;
133
134 for(int32_t i2=symmetric?i1+1:0; i2<n_elem2; ++i2) {
135 const float* p = (symmetric?cache_pos1:cache_pos2)+i2*4;
136 auto x2 = make_vec3(Float4(p[0]), Float4(p[1]), Float4(p[2]));
137 auto near = mag2(x1-x2)<cutoff2;
138 if(near.none()) continue;
139
140 auto my_id2 = Int4((symmetric?cache_id1:cache_id2)[i2]);
141 auto i2_vec = Int4(i2);
142
143 Int4 is_hit = acceptable_id_pair(my_id1,my_id2) & (symmetric
144 ? (i1_vec<i2_vec) & near.cast_int()
145 : near.cast_int());
146 int is_hit_bits = is_hit.movemask();
147
148 // i2_vec and my_id2 is constant, so we don't have to left pack
149 // left_pack requires a read, so do before the writes
150
151 // write out pairs
152 int n_hit = popcnt_nibble(is_hit_bits);
153 i1_vec.left_pack(is_hit_bits).store(cache_edge_indices1+ne, Alignment::unaligned);
154 my_id1.left_pack(is_hit_bits).store(cache_edge_id1 +ne, Alignment::unaligned);
155 i2_vec .store(cache_edge_indices2+ne, Alignment::unaligned);
156 my_id2 .store(cache_edge_id2 +ne, Alignment::unaligned);
157 ne += n_hit;
158 }
159 }
160 cache_n_edge = ne;
161 for(int i=ne; i<round_up(ne,4); ++i) {
162 // we need something sane to fill out the last group of 4 so just duplicate the interactions
163 // with sensitivity 0.
164 cache_edge_indices1[i] = cache_edge_indices1[i-i%4];
165 cache_edge_indices2[i] = cache_edge_indices2[i-i%4]; // just put something sane here
166 }
167 cache_valid = true;
168 // printf("found %i cache edges\n", cache_n_edge);
169 }

202 template<acceptable_id_pair_t acceptable_id_pair>
203 void find_edges(float cutoff,
204 const float* aligned_pos1, const int pos1_stride, int* id1,
205 const float* aligned_pos2, const int pos2_stride, int* id2) {
206 // Timer timer_total("find_edges");
207 ensure_cache_valid<acceptable_id_pair>(cutoff,
208 aligned_pos1, pos1_stride, id1,
209 aligned_pos2, pos2_stride, id2);
210 // Timer timer("pairlist_refine");
211
212 int ne=0;
213 alignas(16) int32_t offset_v[4] = {0,1,2,3};
214 Int4 offset(offset_v);
215 Float4 cutoff2(sqr(cutoff));
216
217 int acceptable = 0;
218 for(int i_edge=0; i_edge<cache_n_edge; i_edge+=4) {
219 auto i1 = Int4(cache_edge_indices1+i_edge);
220 auto i2 = Int4(cache_edge_indices2+i_edge);
221 auto eid1 = Int4(cache_edge_id1+i_edge);
222 auto eid2 = Int4(cache_edge_id2+i_edge);
223
224 Float4 x_diff[4];
225 #pragma unroll
226 for(int j=0; j<4; ++j)
227 x_diff[j] =
228 Float4(aligned_pos1 +pos1_stride*cache_edge_indices1[i_edge+j]) -
229 Float4((symmetric?aligned_pos1:aligned_pos2)+pos2_stride*cache_edge_indices2[i_edge+j]);
230 transpose4(x_diff[0],x_diff[1],x_diff[2],x_diff[3]);
231 auto dist2 = sqr(x_diff[0])+sqr(x_diff[1])+sqr(x_diff[2]);
232
233 acceptable = (dist2<cutoff2).movemask();
234
235 i1 .left_pack_inplace(acceptable);
236 i2 .left_pack_inplace(acceptable);
237 eid1.left_pack_inplace(acceptable);
238 eid2.left_pack_inplace(acceptable);
239
240 int n_acceptable = popcnt_nibble(acceptable);
241 i1 .store(edge_indices1+ne, Alignment::unaligned);
242 i2 .store(edge_indices2+ne, Alignment::unaligned);
243 eid1.store(edge_id1 +ne, Alignment::unaligned);
244 eid2.store(edge_id2 +ne, Alignment::unaligned);
245 ne += n_acceptable;
246 // FIXME it would nice to store the transposed positions for later
247 }
248 // It is possible that some edges were inappropriately declared acceptable even though
249 // they were outside cache_n_edge due to the padding for SSE of 4. Let's fix that.
250 int n_extra = round_up(cache_n_edge,4)-cache_n_edge;
251 int invalid_mask = ((1<<4)-1) & ~((1<<(4-n_extra))-1);
252 n_edge = ne-popcnt_nibble(acceptable&invalid_mask);;
253
254 for(int i=n_edge; i<round_up(n_edge,4); ++i) {
255 edge_indices1[i] = edge_indices1[i-i%4];
256 edge_indices2[i] = edge_indices2[i-i%4]; // just put something sane here
257 }
258 }

445 template<bool param_deriv=false>
446 void compute_edges() {
447 // Copy in the data to packed arrays to ensure contiguity
448 {
449 VecArray posv = pos_node1->output;
450 for(int ne=0; ne<n_elem1; ++ne)
451 store_vec(pos1.get()+ne*n_dim1a, load_vec<n_dim1>(posv, loc1[ne]));
452 }
453 if(!symmetric) {
454 VecArray posv = pos_node2->output;
455 for(int ne=0; ne<n_elem2; ++ne)
456 store_vec(pos2.get()+ne*n_dim2a, load_vec<n_dim2>(posv, loc2[ne]));
457 }
458
459 // First find all the edges
460 {
461 pairlist.template find_edges<IType::acceptable_id_pair>(cutoff,
462 pos1.get(), n_dim1a, id1.get(),
463 (symmetric?pos1:pos2).get(), n_dim2a, (symmetric?id1:id2).get());
464 n_edge = pairlist.n_edge;
465 }
466 // printf("n_edge for n_dim1 %i n_dim2 %i n_elem1 %i n_elem2 %i is %i\n", n_dim1, n_dim2, n_elem1, n_elem2, n_edge);
467
468 // Compute edge values
469 if(param_deriv)
470 edge_param_deriv.clear();
471
472 for(int ne=0; ne<n_edge; ne+=4) {
473 auto i1 = Int4(edge_indices1+ne);
474 auto i2 = Int4(edge_indices2+ne);
475
476 auto t1 = Int4(types1.get(),i1);
477 auto t2 = Int4(types2.get(),i2);
478
479 auto interaction_offset = (t1*Int4(n_type2) + t2)*Int4(n_param);
