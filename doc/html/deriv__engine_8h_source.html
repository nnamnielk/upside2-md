377static double relative_rms_deviation(
378 const std::vector<float> &reference, const std::vector<float> &actual) {
379 if(reference.size() != actual.size())
380 throw std::string("impossible size mismatch ") +
381 std::to_string(reference.size()) + " " + std::to_string(actual.size());
382 double diff_mag2 = 0.;
383 double value1_mag2 = 0.;
384 for(size_t i=0; i<reference.size(); ++i) {
385 diff_mag2 += sqr(reference[i]-actual[i]);
386 value1_mag2 += sqr(reference[i]);
387 }
388 return sqrt(diff_mag2/value1_mag2);
389}
