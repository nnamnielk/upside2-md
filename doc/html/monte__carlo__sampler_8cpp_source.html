29PivotSampler::PivotSampler(const std::string& name, hid_t grp, H5Logger& logger): // Constructor definition
30 MonteCarloSampler(name, PIVOT_MOVE_RANDOM_STREAM, logger),
31 n_layer(h5::get_dset_size(3, grp, "proposal_pot")[0]),
32 n_bin (h5::get_dset_size(3, grp, "proposal_pot")[1]),
33 n_pivot_loc(h5::get_dset_size(2, grp, "pivot_atom")[0]),
34
35 pivot_loc(n_pivot_loc),
36 proposal_prob_cdf(n_layer*n_bin*n_bin)
37{
38 using namespace h5;
39
40 check_size(grp, "proposal_pot", n_layer, n_bin, n_bin);
41 check_size(grp, "pivot_atom", n_pivot_loc, 5);
42 check_size(grp, "pivot_range", n_pivot_loc, 2);
43 check_size(grp, "pivot_restype", n_pivot_loc);
44
45 traverse_dset<2,int>(grp, "pivot_atom", [&](size_t np, size_t na, int x) {pivot_loc[np].rama_atom[na] =x;});
46 traverse_dset<2,int>(grp, "pivot_range", [&](size_t np, size_t j, int x) {pivot_loc[np].pivot_range[j]=x;});
47 traverse_dset<1,int>(grp, "pivot_restype", [&](size_t np, int x) {pivot_loc[np].restype =x;});
48
49 for(auto &p: pivot_loc) {
50 if(p.restype<0 || p.restype>=n_layer) throw std::string("invalid pivot restype");
51 for(int na=0; na<5; ++na) {
52 if(p.pivot_range[0] <= p.rama_atom[na] && p.rama_atom[na] < p.pivot_range[1])
53 throw std::string("pivot_range cannot contain any atoms in pivot_atom ") +
54 std::to_string(p.pivot_range[0]) + " <= " + std::to_string(p.rama_atom[na]) +
55 " < " + std::to_string(p.pivot_range[1]);
56 }
57 }
58
59 traverse_dset<3,float>(grp, "proposal_pot", [&](size_t nl, size_t nx, size_t ny, float x) {proposal_pot.push_back(x);});
60
61 for(int nl=0; nl<n_layer; ++nl) {
62 double sum_prob = 0.;
63 for(int i=0; i<n_bin*n_bin; ++i) {
64 sum_prob += exp(-proposal_pot[nl*n_bin*n_bin + i]);
65 proposal_prob_cdf[nl*n_bin*n_bin + i] = sum_prob;
66 }
67
68 // normalize both the negative log probability and the cdf
69 double inv_sum_prob = 1./sum_prob;
70 double lsum_prob = log(sum_prob);
71 for(int i=0; i<n_bin*n_bin; ++i) {
72 proposal_prob_cdf[nl*n_bin*n_bin + i] *= inv_sum_prob;
73 proposal_pot[nl*n_bin*n_bin + i] += lsum_prob;
74 }
75
76 proposal_prob_cdf[(nl+1)*n_bin*n_bin-1] = 1.f; // ensure no rounding error here
77 }

80void PivotSampler::propose_random_move(float* delta_lprob,
81 RandomGenerator& random, VecArray pos) const {
82 Timer timer(std::string("random_pivot"));
83 float4 random_values = random.uniform_open_closed();
84
85 // pick a random pivot location
86 int loc = int(n_pivot_loc * random_values.z());
87 if(loc == n_pivot_loc) loc--; // this may occur due to rounding
88 auto p = pivot_loc[loc];
89
90 // select a bin
91 float cdf_value = random_values.w();
92
93 int pivot_bin = std::lower_bound(
94 proposal_prob_cdf.data()+ p.restype *n_bin*n_bin,
95 proposal_prob_cdf.data()+(p.restype+1)*n_bin*n_bin,
96 cdf_value) - (proposal_prob_cdf.data()+p.restype*n_bin*n_bin);
97 float new_lprob = proposal_pot[p.restype*n_bin*n_bin + pivot_bin];
98
99 int phi_bin = pivot_bin/n_bin;
100 int psi_bin = pivot_bin%n_bin;
101
102 // now pick a random location in that bin
103 // Note the half-bin shift because we want the bin center of the left-most bin at 0
104 float2 new_rama = (2.f*M_PI_F/n_bin)*make_vec2(phi_bin+random_values.x()-0.5f, psi_bin+random_values.y()-0.5f) - M_PI_F;
105
106 // find deviation from old rama
107 float3 d1,d2,d3,d4;
108 float3 prevC = load_vec<3>(pos, p.rama_atom[0]);
109 float3 N = load_vec<3>(pos, p.rama_atom[1]);
110 float3 CA = load_vec<3>(pos, p.rama_atom[2]);
111 float3 C = load_vec<3>(pos, p.rama_atom[3]);
112 float3 nextN = load_vec<3>(pos, p.rama_atom[4]);
113
114 float2 old_rama = make_vec2(
115 dihedral_germ(prevC,N,CA,C, d1,d2,d3,d4),
116 dihedral_germ(N,CA,C,nextN, d1,d2,d3,d4));
117
118 // reverse the half-bin shift
119 int old_phi_bin = (old_rama.x()+M_PI_F) * (0.5f/M_PI_F) * n_bin + 0.5f;
120 int old_psi_bin = (old_rama.y()+M_PI_F) * (0.5f/M_PI_F) * n_bin + 0.5f;
121 old_phi_bin = old_phi_bin>=n_bin ? 0 : old_phi_bin; // enforce periodicity
122 old_psi_bin = old_psi_bin>=n_bin ? 0 : old_psi_bin;
123 float old_lprob = proposal_pot[(p.restype*n_bin + old_phi_bin)*n_bin + old_psi_bin];
124
125 // apply rotations
126 float3 phi_origin = CA;
127 float3 psi_origin = C;
128
129 float2 delta_rama = new_rama - old_rama;
130 float phi_U[9]; axis_angle_to_rot(phi_U, delta_rama.x(), normalized(CA-N ));
131 float psi_U[9]; axis_angle_to_rot(psi_U, delta_rama.y(), normalized(C -CA));
132
133 {
134 auto y = load_vec<3>(pos, p.rama_atom[3]); // C
135 float3 after_psi = psi_origin + apply_rotation(psi_U, y -psi_origin); // unnecessary but harmless
136 float3 after_phi = phi_origin + apply_rotation(phi_U, after_psi-phi_origin);
137 store_vec(pos, p.rama_atom[3], after_phi);
138 }
139
140 {
141 auto y = load_vec<3>(pos, p.rama_atom[4]); // nextN
142 float3 after_psi = psi_origin + apply_rotation(psi_U, y -psi_origin);
143 float3 after_phi = phi_origin + apply_rotation(phi_U, after_psi-phi_origin);
144 store_vec(pos, p.rama_atom[4], after_phi);
145 }
146
147 for(int na=p.pivot_range[0]; na<p.pivot_range[1]; ++na) {
148 auto y = load_vec<3>(pos, na);
149 float3 after_psi = psi_origin + apply_rotation(psi_U, y -psi_origin);
150 float3 after_phi = phi_origin + apply_rotation(phi_U, after_psi-phi_origin);
151 store_vec(pos, na, after_phi);
152 }
153
154 *delta_lprob = new_lprob - old_lprob;
155}

203void JumpSampler::propose_random_move(float* delta_lprob,
204 RandomGenerator& random, VecArray pos) const {
205 Timer timer(std::string("random_jump"));
206
207 // pick jump move type: translation or rotation
208 float4 rand_type_val = random.uniform_open_closed();
209 int jump_move_type = int(2 * rand_type_val.x());
210
211 // pick a random jump chain
212 int chain = int(n_jump_chains * rand_type_val.w());
213 if(chain == n_jump_chains) chain--; // this may occur due to rounding
214 const auto& j = jump_chains[chain];
215
216 if (jump_move_type == 0) { // translation
217 // pick a random jump translation
218 float3 rand_disp_val = j.sigma_trans/sqrtf(3.f) * random.normal3();
219
220 // apply displacement
221 for (int na = j.first_atom; na < j.next_first; na++) {
222 float3 pos_na = load_vec<3>(pos, na);
223 float3 new_pos_na = rand_disp_val + pos_na;
224 store_vec(pos, na, new_pos_na);
225 }
226 }
227 else { // rotation
228 // pick a random jump rotation angle and axis unit vector. Create rotation matrix
229 float4 rand_rot_variates = random.normal();
230 float rand_rot_angle = j.sigma_rot * rand_rot_variates[0];
231 float3 rand_rot_axis = extract<1,4>(rand_rot_variates);
232 rand_rot_axis /= mag(rand_rot_axis)+1e-16f; // 1e-16 is paranoia against division by zero
233
234 float U[9]; axis_angle_to_rot(U, rand_rot_angle, rand_rot_axis);
235
236 // get CoM
237 float3 com = make_vec3(0.f, 0.f, 0.f);
238 for (int na = j.first_atom; na < j.next_first; na++)
239 com += load_vec<3>(pos, na);
240 com *= 1.f/(j.next_first-j.first_atom);
241
242 // apply rotation about com
243 for (int na = j.first_atom; na < j.next_first; na++) {
244 float3 pos_na = load_vec<3>(pos, na);
245 float3 new_pos_na = com + apply_rotation(U, pos_na-com);
246 store_vec(pos, na, new_pos_na);
247 }
248 }
249
250 *delta_lprob = prob_mod; //0.f;
251}

255void MonteCarloSampler::monte_carlo_step(
256 uint32_t seed,
257 uint64_t round,
258 const float temperature,
259 DerivEngine& engine)
260{
261 RandomGenerator random(seed, stream_id, 0, round);
262
263 auto &pos = engine.pos->output;
264 VecArrayStorage pos_copy(pos);
265 float delta_lprob;
266
267 engine.compute(PotentialAndDerivMode);
268 float old_potential = engine.potential;
269
270 propose_random_move(&delta_lprob, random, pos);
271
272 engine.compute(PotentialAndDerivMode);
273 float new_potential = engine.potential;
274
275 float lboltz_diff = delta_lprob - (1.f/temperature) * (new_potential-old_potential);
276 move_stats.n_attempt++;
277
278 if(lboltz_diff >= 0.f || expf(lboltz_diff) >= random.uniform_open_closed().x()) {
279 move_stats.n_success++;
280 } else {
281 // If we reject the move, we must reverse it
282 copy(pos_copy, pos);
283 }
284}
