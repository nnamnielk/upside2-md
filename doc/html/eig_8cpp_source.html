56S house(
57 int n,
58 S* x) // size (n,), replaced by v, v[0] = |x|
59{
60 S beta,mu,s;
61 S sigma2 = 1e-20f;
62
63 for(int i=1; i<n; ++i) sigma2 += x[i]*x[i];
64
65 mu = sqrtf(x[0]*x[0] + sigma2); // |x|
66 s = ternary(zero<S>()<x[0], -sigma2*rcp(x[0]+mu), x[0]-mu);
67 beta = two<S>()*s*s * rcp(sigma2+s*s);
68
69 x[0] = mu;
70 for(int i=1; i<n; ++i) x[i] *= rcp(s);
71
72 return beta;
73}

75void givens(S *c, S *s, S a, S b)
76{
77 S inv_r = rsqrt(a*a+b*b);
78
79 auto trivial = b==zero<S>();
80 *c = ternary(trivial, one <S>(), a*inv_r);
81 *s = ternary(trivial, zero<S>(), -b*inv_r);
82}

86int r(int i, int j) {
87 int ii = (i<j) ? i : j;
88 int jj = (i<j) ? j : i;
89
90 switch(ii) {
91 case 0:
92 return (0-0) + jj;
93 case 1:
94 return (4-1) + jj;
95 case 2:
96 return (7-2) + jj;
97 case 3:
98 return (9-3) + jj;
99 default:
100 return 1000; // error
101 }

105void
106symmetric_tridiagonalize_4x4(
107 S* restrict beta, // length 2 beta sequence of Householder sequence
108 S* restrict A) // upper triangle of 4x4 matrix, 10 elements
109{
110 S p[3];
111 S w[3];
112
113 for(int k=0; k<4-2; ++k) {
114 int m = 4-(k+1);
115 beta[k] = house(m, A+r(k,k+1)); // overwrite with householder
116
117 #define v(j) ((j)==0 ? one<S>() : A[r(k,k+1+(j))])
118 for(int i=0; i<m; ++i) {
119 p[i] = zero<S>();
120 for(int j=0; j<m; ++j)
121 p[i] += A[r(i+k+1,j+k+1)] * v(j);
122 p[i] *= beta[k];
123 }
124
125 S p_dot_v = zero<S>();
126 for(int i=0; i<m; ++i) p_dot_v += p[i]*v(i);
127 for(int i=0; i<m; ++i) w[i] = p[i] - (half<S>()*beta[k]*p_dot_v) * v(i);
128
129 for(int i=0; i<m; ++i)
130 for(int j=i; j<m; ++j)
131 A[r(i+k+1,j+k+1)] -= v(i)*w[j] + v(j)*w[i];
132 #undef v
133 }
134}

137void
138unpack_tridiagonalize_4x4(
139 S * restrict d, // length 4
140 S * restrict u, // length 3
141 S * restrict rot_, // size 4x4 = 16, nonsymmetric
142 S * restrict beta, // length 2
143 S * restrict A) // symmetric_4x4, length 10
144{
145 #define rot(i,j) (rot_[(i)*4+(j)])
146 for(int i=0; i<4; ++i) d[i] = A[r(i,i)];
147 for(int i=0; i<3; ++i) u[i] = A[r(i,i+1)];
148
149 // first update with the symmetric part
150 for(int i=0; i<4; ++i) for(int j=i; j<4; ++j) rot(i,j) = (i==j); // identity matrix to start
151
152 // unpack Householder reflection vectors
153 #define v0(j) ((j)==0 ? one<S>() : A[r(0,1+(j))])
154 #define v1(j) ((j)==0 ? one<S>() : A[r(1,2+(j))])
155
156 // first Householder reflection operates on lower left 3x3 block
157 for(int i=1; i<4; ++i)
158 for(int j=i; j<4; ++j)
159 rot(i,j) -= beta[0] * (v0(i-1)*v0(j-1));
160
161 // second Householder reflection operates on the lowest 2x2 block
162 for(int i=2; i<4; ++i)
163 for(int j=i; j<4; ++j)
164 rot(i,j) -= beta[1] * (v1(i-2)*v1(j-2));
165
166 // fill in using symmetry
167 for(int i=0; i<4; ++i) for(int j=i+1; j<4; ++j) rot(j,i) = rot(i,j);
168
169 // now fill in non-symmetric Householder interaction (note leading zero for v1)
170 // fills in lower left 2x3 block
171 S coeff = beta[0]*beta[1] * (v0(1)*v1(0) + v0(2)*v1(1));
172 for(int i=2; i<4; ++i)
173 for(int j=1; j<4; ++j)
174 rot(i,j) += coeff * v1(i-2)*v0(j-1);
175 #undef rot
176 #undef v0
177 #undef v1
178}

183void
184implicit_symm_QR_step_4x4(
185 int n, // length of diagonal
186 S * restrict d, // diagonal
187 S * restrict u, // upper superdiagonal
188 S * restrict rot_) // rotation to accumulate Givens rotations
189 // must be at least nx4-sized
190{
191 S dval = half<S>()*(d[n-2] - d[n-1]) + S(1e-20f); // paranoia against NaN
192 S mu = d[n-1] - u[n-2]*u[n-2]*rcp(dval+copysignf(sqrtf(dval*dval + u[n-2]*u[n-2]),dval));
193
194 S x = d[0] - mu;
195 S z = u[0];
196
197 for(int k=0; k<n-1; ++k){
198 S c,s;
199 givens(&c,&s,x,z);
200
201 /* accumulate Givens rotation */
202 for(int j=0; j<4; ++j) {
203 S t1 = rot(k ,j);
204 S t2 = rot(k+1,j);
205 rot(k ,j) = c*t1 - s*t2;
206 rot(k+1,j) = s*t1 + c*t2;
207 }
208
209 /* compute the T(k-1,:) row */
210 if(k > 0) u[k-1] = c*x - s*z;
211
212 /* mix the T(k,k),T(k,k+1),T(k+1,k+1) block */
213 S T00 = d[k ];
214 S T11 = d[k+1];
215 S T01 = u[k ];
216
217 d[k ] = T00*c*c - T01*two<S>()*c*s + T11*s*s;
218 d[k+1] = T00*s*s + T01*two<S>()*c*s + T11*c*c;
219 u[k ] = (T00-T11)*c*s + T01*(c*c-s*s);
220 x = u[k];
221
222 /* compute the T(k+2,:) row */
223 if(k<n-2) {
224 z = -u[k+1]*s;
225 u[k+1] *= c;
226 }
227 }
228}
229#undef rot

232int __attribute__ ((noinline))
233symm_QR_4x4(
234 S* restrict d, // will contain eigenvalues, length 4
235 S* restrict rot, // will contain eigenvectors in rows, size 4x4
236 S* restrict A, // symmetric 4x4 matrix, 10 elements, overwritten
237 S tol, // relative tolerance for off-diagonal entries (suggest something like 1e-5)
238 int max_iter)
239{
240 const int n = 4;
241
242 S beta[2];
243 symmetric_tridiagonalize_4x4(beta, A);
244
245 S u[3];
246 unpack_tridiagonalize_4x4(d, u, rot, beta, A);
247
248 for(int k=0; k<max_iter; ++k) {
249 // exactly zero off-diagonal elements that are nearly zero
250 for(int i=0; i<n-1; ++i) {
251 auto small_u = fabsf(u[i]) <= tol * (fabsf(d[i]) + fabsf(d[i+1]));
252 u[i] = ternary(small_u, zero<S>(), u[i]);
253 }
254
255 // find largest diagonal lower left subblock
256 int q;
257 if (any(u[2] != zero<S>())) q = 0;
258 else if (any(u[1] != zero<S>())) q = 1;
259 else if (any(u[0] != zero<S>())) q = 2;
260 else return k; // q=3 and the matrix is diagonal and we are done
261
262 // p is start of largest unreduced triangular submatrix adjacent to the q region
263 // unreduced means no entries in the upper triangle are zero
264 int p;
265 for(p=n-q-1; p>0; --p) {
266 if(none(u[p-1] != zero<S>())) break;
267 }
268
269 // perform QR iteration and accumulate into rot
270 implicit_symm_QR_step_4x4(n-q-p, d+p, u+p, rot+4*p);
271 }
272 return -1; // non-convergence
273}

388 virtual void propagate_deriv() {
389 Timer timer(string("affine_alignment_deriv"));
390 float* pos_sens = pos.sens.x.get();
391
392 for(int ng=0; ng<n_group; ++ng) {
393 const auto& p = params[ng];
394
395 const Float4* restrict evals = evals_storage.get() + ng* 4;
396 const Float4* restrict evecs = evecs_storage.get() + ng*16;
397
398 // compute inverse eigenvalue differences
399 S inv_evals[4];
400 for(int j=1; j<4; ++j) inv_evals[j] = rcp(evals[0]-evals[j]);
401
402 // push back torque to affine derivatives (multiply by quaternion)
403 // the torque is in the tangent space of the rotated frame
404 // to act on a tangent in the affine space, I need to push that tangent into the rotated space
405 // this means a right multiply by the quaternion itself
406
407 // evecs[0] is the rotation quaternion
408 S sens3 [3] = {Float4(&sens(0,4*ng+0)), Float4(&sens(0,4*ng+1)),Float4(&sens(0,4*ng+2))};
409 S torque [3] = {Float4(&sens(0,4*ng+3)), Float4(&sens(4,4*ng+0)),Float4(&sens(4,4*ng+1))};
410 S padding[2] = {Float4(&sens(4,4*ng+2)), Float4(&sens(4,4*ng+3))};
411
412 transpose4(sens3 [0],sens3 [1],sens3 [2],torque [0]);
413 transpose4(torque[1],torque[2],padding[0],padding[1]);
414
415 S quat_sens[4] = {
416 two<S>()*(-torque[0]*evecs[1] - torque[1]*evecs[2] - torque[2]*evecs[3]),
417 two<S>()*( torque[0]*evecs[0] + torque[1]*evecs[3] - torque[2]*evecs[2]),
418 two<S>()*( torque[1]*evecs[0] + torque[2]*evecs[1] - torque[0]*evecs[3]),
419 two<S>()*( torque[2]*evecs[0] + torque[0]*evecs[2] - torque[1]*evecs[1])};
420
421 S quat_sens_diag_basis[4] = {zero<S>(),zero<S>(),zero<S>(),zero<S>()};
422 for(int d=0; d<4; ++d) for(int i=0; i<4; ++i) quat_sens_diag_basis[d] += quat_sens[i]*evecs[d*4+i];
423
424 // The derivative of an eigenvector is given by perturbation theory as
425 // a linear combination of the other eigenvectors weighted by interaction
426 // matrix elements and the difference of eigenvalues
427
428 // I will assume that the largest eigenvalue is nondegenerate, as would be expected
429 // for a reasonable alignment to a nearly rigid structure
430
431 // unsafe macro for perturbation term
432 #define t(i,j) (f[r(i,j)] * ((i==j) \
433 ? evecs[k*4+i]*evecs[0*4+j] \
434 : evecs[k*4+i]*evecs[0*4+j]+evecs[k*4+j]*evecs[0*4+i]))
435 #define perturb(ret, f00,f01,f02,f03,f11,f12,f13,f22,f23,f33) do { \
436 const S f[10] = {f00,f01,f02,f03,f11,f12,f13,f22,f23,f33}; \
437 for(int k=1; k<4; ++k) { \
438 S c = inv_evals[k]*(t(0,0)+t(0,1)+t(0,2)+t(0,3) \
439 +t(1,1)+t(1,2)+t(1,3) \
440 +t(2,2)+t(2,3) \
441 +t(3,3)); \
442 (ret) += c*quat_sens_diag_basis[k]; \
443 } \
444 } while(0)
445
446
447 for(int na=0; na<3; ++na) {
448 Vec<3,S> deriv = S(1.f/3.f)*make_vec3(sens3[0],sens3[1],sens3[2]);
449 auto g = make_vec3(Float4(p.ref_geom[na][0]), Float4(p.ref_geom[na][1]), Float4(p.ref_geom[na][2]));
450
451 // quaternion rotation derivative
452 perturb(deriv[0], g[0], zero<S>(), g[2],-g[1],
453 g[0], g[1], g[2],
454 -g[0], zero<S>(),
455 -g[0]);
456
457 perturb(deriv[1], g[1],-g[2], zero<S>(), g[0],
458 -g[1], g[0], zero<S>(),
459 g[1], g[2],
460 -g[1]);
461
462 perturb(deriv[2], g[2], g[1], -g[0], zero<S>(),
463 -g[2], zero<S>(), g[0],
464 -g[2], g[1],
465 g[2]);
466
467 aligned_scatter_update_vec_destructive(pos_sens, Int4(p.atom_offsets[na]), deriv);
468 }
469 }
470 }
