92 void execute_random_pivot(float* delta_lprob,
93 uint32_t seed, uint64_t n_round, VecArray pos) const {
94 Timer timer(std::string("random_pivot"));
95 RandomGenerator random(seed, PIVOT_MOVE_RANDOM_STREAM, 0, n_round);
96 float4 random_values = random.uniform_open_closed();
97
98 // pick a random pivot location
99 int loc = int(n_pivot_loc * random_values.z());
100 if(loc == n_pivot_loc) loc--; // this may occur due to rounding
101 auto p = pivot_loc[loc];
102
103 // select a bin
104 float cdf_value = random_values.w();
105
106 int pivot_bin = std::lower_bound(
107 proposal_prob_cdf.data()+ p.restype *n_bin*n_bin,
108 proposal_prob_cdf.data()+(p.restype+1)*n_bin*n_bin,
109 cdf_value) - (proposal_prob_cdf.data()+p.restype*n_bin*n_bin);
110 float new_lprob = proposal_pot[p.restype*n_bin*n_bin + pivot_bin];
111
112 int phi_bin = pivot_bin/n_bin;
113 int psi_bin = pivot_bin%n_bin;
114
115 // now pick a random location in that bin
116 // Note the half-bin shift because we want the bin center of the left-most bin at 0
117 float2 new_rama = (2.f*M_PI_F/n_bin)*make_vec2(phi_bin+random_values.x()-0.5f, psi_bin+random_values.y()-0.5f) - M_PI_F;
118
119 // find deviation from old rama
120 float3 d1,d2,d3,d4;
121 float3 prevC = load_vec<3>(pos, p.rama_atom[0]);
122 float3 N = load_vec<3>(pos, p.rama_atom[1]);
123 float3 CA = load_vec<3>(pos, p.rama_atom[2]);
124 float3 C = load_vec<3>(pos, p.rama_atom[3]);
125 float3 nextN = load_vec<3>(pos, p.rama_atom[4]);
126
127 float2 old_rama = make_vec2(
128 dihedral_germ(prevC,N,CA,C, d1,d2,d3,d4),
129 dihedral_germ(N,CA,C,nextN, d1,d2,d3,d4));
130
131 // reverse the half-bin shift
132 int old_phi_bin = (old_rama.x()+M_PI_F) * (0.5f/M_PI_F) * n_bin + 0.5f;
133 int old_psi_bin = (old_rama.y()+M_PI_F) * (0.5f/M_PI_F) * n_bin + 0.5f;
134 old_phi_bin = old_phi_bin>=n_bin ? 0 : old_phi_bin; // enforce periodicity
135 old_psi_bin = old_psi_bin>=n_bin ? 0 : old_psi_bin;
136 float old_lprob = proposal_pot[(p.restype*n_bin + old_phi_bin)*n_bin + old_psi_bin];
137
138 // apply rotations
139 float3 phi_origin = CA;
140 float3 psi_origin = C;
141
142 float2 delta_rama = new_rama - old_rama;
143 float phi_U[9]; axis_angle_to_rot(phi_U, delta_rama.x(), normalized(CA-N ));
144 float psi_U[9]; axis_angle_to_rot(psi_U, delta_rama.y(), normalized(C -CA));
145
146 {
147 auto y = load_vec<3>(pos, p.rama_atom[3]); // C
148 float3 after_psi = psi_origin + apply_rotation(psi_U, y -psi_origin); // unnecessary but harmless
149 float3 after_phi = phi_origin + apply_rotation(phi_U, after_psi-phi_origin);
150 store_vec(pos, p.rama_atom[3], after_phi);
151 }
152
153 {
154 auto y = load_vec<3>(pos, p.rama_atom[4]); // nextN
155 float3 after_psi = psi_origin + apply_rotation(psi_U, y -psi_origin);
156 float3 after_phi = phi_origin + apply_rotation(phi_U, after_psi-phi_origin);
157 store_vec(pos, p.rama_atom[4], after_phi);
158 }
159
160 for(int na=p.pivot_range[0]; na<p.pivot_range[1]; ++na) {
161 auto y = load_vec<3>(pos, na);
162 float3 after_psi = psi_origin + apply_rotation(psi_U, y -psi_origin);
163 float3 after_phi = phi_origin + apply_rotation(phi_U, after_psi-phi_origin);
164 store_vec(pos, na, after_phi);
165 }
166
167 *delta_lprob = new_lprob - old_lprob;
168 }

170 void pivot_monte_carlo_step(
171 uint32_t seed,
172 uint64_t round,
173 const float temperature,
174 DerivEngine& engine)
175 {
176 auto &pos = engine.pos->output;
177 VecArrayStorage pos_copy(pos);
178 float delta_lprob;
179
180 engine.compute(PotentialAndDerivMode);
181 float old_potential = engine.potential;
182
183 execute_random_pivot(&delta_lprob, seed, round, pos);
184
185 engine.compute(PotentialAndDerivMode);
186 float new_potential = engine.potential;
187
188 float lboltz_diff = delta_lprob - (1.f/temperature) * (new_potential-old_potential);
189 RandomGenerator random(seed, PIVOT_MONTE_CARLO_RANDOM_STREAM, 0, round);
190 pivot_stats.n_attempt++;
191
192 if(lboltz_diff >= 0.f || expf(lboltz_diff) >= random.uniform_open_closed().x()) {
193 pivot_stats.n_success++;
194 } else {
195 // If we reject the pivot, we must reverse it
196 copy(pos_copy, pos);
197 }
198 }
