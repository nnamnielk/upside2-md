15inline Affine3f make_tab(float phi, float theta, float bond)
16{
17 Affine3f out;
18 float cp(cos(phi)), sp(sin(phi));
19 float ct(cos(theta)), st(sin(theta));
20 float l(bond);
21
22 out(0,0)= -ct; out(0,1)= -st; out(0,2)= 0; out(0,3)= -l*ct;
23 out(1,0)= cp*st; out(1,1)= -cp*ct; out(1,2)= -sp; out(1,3)= l*cp*st;
24 out(2,0)= sp*st; out(2,1)= -sp*ct; out(2,2)= cp; out(2,3)= l*sp*st;
25 // out(3,0)= 0; out(3,1)= 0; out(3,2)= 0; out(3,3)= 1;
26
27 return out;
28}
