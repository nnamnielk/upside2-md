22static float dihedral(
23 Vector3f r1, Vector3f r2, Vector3f r3, Vector3f r4)
24 // Formulas and notation taken from Blondel and Karplus, 1995
25{
26 Vector3f F = r1-r2;
27 Vector3f G = r2-r3;
28 Vector3f H = r4-r3;
29
30 Vector3f A = F.cross(G);
31 Vector3f B = H.cross(G);
32 Vector3f C = B.cross(A);
33
34 return atan2(C.dot(G), A.dot(B) * G.norm());
35}
