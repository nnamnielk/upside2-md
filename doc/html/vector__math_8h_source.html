699// Sigmoid-like function that has zero derivative outside (-1/sharpness,1/sharpness)
700// This function is 1 for large negative values and 0 for large positive values
701template <typename S>
702inline Vec<2,S> compact_sigmoid(const S& x, const S& sharpness) {
703 // FIXME this sigmoid is both narrower and reversed direction from a normal sigmoid
704#ifdef NONCOMPACT_SIGMOID
705 // factor of three makes the slopes even at the origin
706 auto z = sigmoid(-3.f*x*sharpness);
707 return make_vec2(z.x(), -3.f*z.y()*sharpness);
708#else
709 S y = x*sharpness;
710 Vec<2,S> z = make_vec2(S(0.25f)*(y+S(2.f))*(y-S(1.f))*(y-one<S>()), (sharpness*S(0.75f))*(sqr(y)-one<S>()));
711
712 auto too_big = S( 1.f)<y;
713 auto too_small = y<S(-1.f);
714
715 // apply cutoffs
716 z.x() = ternary(too_small, one<S>(), ternary(too_big, zero<S>(), z.x()));
717 z.y() = ternary(too_small | too_big, zero<S>(), z.y());
718 return z;
719#endif
720}

722inline float compact_sigmoid_cutoff(float sharpness) {
723#ifdef NONCOMPACT_SIGMOID
724 return 2.f/sharpness; // 0.002
725#else
726 return 1.f/sharpness; // exactly 0.
727#endif
728}

765template <typename V> // parameterize on vector type (probably Vec<3> or Float4)
766static typename V::scalar_t dihedral_germ(
767 V r1, V r2, V r3, V r4,
768 V &d1, V &d2, V &d3, V &d4)
769 // Formulas and notation taken from Blondel and Karplus, 1995
770{
771 typedef typename V::scalar_t S; // scalar type associated to vector type
772
773 V F = r1-r2;
774 V G = r2-r3;
775 V H = r4-r3;
776
777 V A = cross(F,G);
778 V B = cross(H,G);
779 V C = cross(B,A);
780
781 S inv_Amag2 = inv_mag2(A);
782 S inv_Bmag2 = inv_mag2(B);
783
784 S Gmag2 = mag2(G);
785 S inv_Gmag = rsqrt(Gmag2);
786 S Gmag = Gmag2 * inv_Gmag;
787
788 d1 = -Gmag * inv_Amag2 * A;
789 d4 = Gmag * inv_Bmag2 * B;
790
791 V f_mid = dot(F,G)*inv_Amag2*inv_Gmag * A - dot(H,G)*inv_Bmag2*inv_Gmag * B;
792
793 d2 = -d1 + f_mid;
794 d3 = -d4 - f_mid;
795
796 return atan2f(extract_float(dot(C,G)), extract_float(dot(A,B) * Gmag));
797}

810inline Vec<D,Float4> aligned_gather_vec(const float* data, const Int4& offsets) {
811 Vec<D,Float4> ret;
812

815 const float* p2 = data+offsets.z();

819

822 ret[d ] = Float4(p0+d);

829 (d+1<D ? ret[d+1] : extra[0]),
