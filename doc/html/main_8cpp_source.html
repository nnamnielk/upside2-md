42static void abort_like_handler(int signal) {

44 // This is a signal handler called in response to things like
45 // Ctrl-C. These functions are very special, and there are very
46 // few things you are allowed to do in such a function without
47 // causing problems. Basically, just set a global variable of type
48 // volatile sig_atomic_t and then notice the flag is set later in
49 // the code. Even most forms of exiting the program are disallowed
50 // in a signal handler. Please do not edit this function unless
51 // you have read extensively about signal handlers, especially for
52 // multithreaded code.
53
54 // This signal handler is intended for signals that indicate the user
55 // or OS wishes the program to terminate but after possibly writing any
56 // remaining state. There is some danger that a later SIGKILL could
57 // corrupt the file being written to dump the last small amount of
58 // state. I think this is a risk worth taking.
59
60 // Note that repeated Ctrl-C will not terminate the program if it is
61 // still writing to the (possibly hanging) filesystem. This is also
62 // annoying, but the user can terminate with SIGKILL.
63
64 received_signal = signal;

134vector<string> split_string(const string& src, const string& sep) {
135 vector<string> ret;
136
137 for(auto curr_pos = begin(src); curr_pos-begin(src) < ptrdiff_t(src.size()); ) {
138 auto next_match_end = search(curr_pos, end(src), begin(sep), end(sep));
139 ret.emplace_back(curr_pos, next_match_end);
140 curr_pos = next_match_end + sep.size();
141 }

170 set.emplace_back();

176 s.sys1 = stoi_strict(p[0]);

193 for(auto& sw: ss) {
194 if(systems_in_set.count(sw.sys1) ||
195 systems_in_set.count(sw.sys2) ||
196 sw.sys1==sw.sys2) {
197 throw string("Overlapping indices in swap set. "
198 "No replica index can appear more than once in a swap set. "
199 "You probably (but maybe not; I didn't look that closely) need more "
200 "swap sets to get non-overlapping pairs.");
201 }
202 systems_in_set.insert(sw.sys1);
203 systems_in_set.insert(sw.sys2);
204
205 participating_swaps[sw.sys1].push_back(&sw);
206 participating_swaps[sw.sys2].push_back(&sw);
207 }
208 }

210 // enable logging of replica events
211 for(int ns: range(n_system)) {
212 auto logger = systems[ns].logger;

243 void attempt_swaps(uint32_t seed, uint64_t round, vector<System>& systems, int exchange_criterion ) {
244 int n_system = systems.size();
245
246 vector<float> beta;
247 for(auto &sys: systems) beta.push_back(1.f/sys.temperature);
248
249 // compute the boltzmann factors for everyone
250 auto compute_log_boltzmann = [&]() {
251 vector<float> result(n_system);
252 for(int i=0; i<n_system; ++i) {
253 systems[i].engine.compute(PotentialAndDerivMode);
254 result[i] = -beta[i]*systems[i].engine.potential;
255 }
256 return result;
257 };
258
259 // swap coordinates and the associated system indices
260 auto coord_swap = [&](int ns1, int ns2) {
261 swap(systems[ns1].engine.pos->output, systems[ns2].engine.pos->output);
262 swap(replica_indices[ns1], replica_indices[ns2]);
263 };
264
265 RandomGenerator random(seed, REPLICA_EXCHANGE_RANDOM_STREAM, 0u, round);
266
267 int swap_select = n_attempt_swaps%n_swap_sets;
268
269 for(auto& set: swap_sets) {
270 // FIXME the first energy computation is unnecessary if we are not on the first swap set
271 // It is important that the energy is computed more than once in case
272 // we are doing Hamiltonian parallel tempering rather than
273 // temperature parallel tempering
274
275 if (set[0].set_id != swap_select)
276 continue;
277
278 if (exchange_criterion == 0) {
279 auto old_lboltz = compute_log_boltzmann();
280 for(auto& swap_pair: set) coord_swap(swap_pair.sys1, swap_pair.sys2);
281 auto new_lboltz = compute_log_boltzmann();
282
283 // reverse all swaps that should not occur by metropolis criterion
284 for(auto& swap_pair: set) {
285 auto s1 = swap_pair.sys1;
286 auto s2 = swap_pair.sys2;
287 swap_pair.n_attempt++;
288
289 float lboltz_diff = (new_lboltz[s1] + new_lboltz[s2]) - (old_lboltz[s1]+old_lboltz[s2]);
290 // If we reject the swap, we must reverse it
291 if(lboltz_diff < 0.f && expf(lboltz_diff) < random.uniform_open_closed().x()) {
292 coord_swap(s1,s2);
293 } else {
294 swap_pair.n_success++;
295 }
296 }
297 }
298 else if (exchange_criterion == 1) {
299 auto lboltz = compute_log_boltzmann();
300
301 // reverse all swaps that should not occur by metropolis criterion
302 for(auto& swap_pair: set) {
303 auto s1 = swap_pair.sys1;
304 auto s2 = swap_pair.sys2;
305 swap_pair.n_attempt++;
306 float lboltz_diff = lboltz[s2] - lboltz[s1];
307 // If we reject the swap, we must reverse it
308 if(lboltz_diff >= 0.f || expf(lboltz_diff) >= random.uniform_open_closed().x()) {
309 coord_swap(s1,s2);
310 swap_pair.n_success++;
311 }
312 }
313 }
314 }
315 n_attempt_swaps += 1;
316 }
