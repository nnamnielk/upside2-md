16 vector<float> copy_eigen_to_stl(const ArrayXXf& mat) {
17 // Output is row-major order, which is the opposite of the Eigen default
18 vector<float> ret(mat.rows()*mat.cols());

63 void update_transition_matrix_from_transition_energy() {
64 // Offset by the expected value of the energy to increase numerical stability
65 float e_min = transition_energy.minCoeff();
66 energy_offset = (transition_energy.array() * (e_min - transition_energy.array()).exp()).sum() /
67 ((e_min - transition_energy.array()).exp()).sum();
68 transition_matrix = (energy_offset-transition_energy).exp();
