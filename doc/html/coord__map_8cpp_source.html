422struct BackboneFeaturizer : public CoordNode
423{
424 CoordNode& rama;
425 CoordNode& hbond;
426
427 struct Param {
428 int rama_idx;
429 int donor_idx;
430 int acceptor_idx;
431 };
432 vector<Param> params;
433
434 BackboneFeaturizer(hid_t grp, CoordNode& rama_, CoordNode& hbond_):
435 CoordNode(get_dset_size(1, grp, "rama_idx")[0], 6),
436 rama(rama_),
437 hbond(hbond_),
438 params(n_elem)
439 {
440 check_size(grp, "rama_idx", n_elem);
441 check_size(grp, "hbond_idx", n_elem, 2);
442
443 traverse_dset<1,int>(grp, "rama_idx", [&](size_t ne, int x){params[ne].rama_idx=x;});
444
445 // -1 is special code to indicate no hbond donor or acceptor
446 traverse_dset<2,int>(grp, "hbond_idx", [&](size_t ne, size_t is_acceptor, int x){
447 if(is_acceptor) params[ne].acceptor_idx=x;
448 else params[ne].donor_idx =x;});
449 }
450
451 virtual void compute_value(ComputeMode mode) override {
452 VecArray ramac = rama.output;
453 VecArray hbondc = hbond.output;
454
455 for(int ne=0; ne<n_elem; ++ne) {
456 auto& p = params[ne];
457
458 float phi = ramac(0,p.rama_idx);
459 float psi = ramac(1,p.rama_idx);
460 float don_hb = p.donor_idx ==-1 ? 0.f : hbondc(6,p.donor_idx);
461 float acc_hb = p.acceptor_idx==-1 ? 0.f : hbondc(6,p.acceptor_idx);
462
463 output(0,ne) = sin(phi);
464 output(1,ne) = cos(phi);
465 output(2,ne) = sin(psi);
466 output(3,ne) = cos(psi);
467 output(4,ne) = don_hb;
468 output(5,ne) = acc_hb;
469 }
470 }
471
472 virtual void propagate_deriv() override {
473 VecArray rama_s = rama.sens;
474 VecArray hbond_s = hbond.sens;
475
476 for(int ne=0; ne<n_elem; ++ne) {
477 auto& p = params[ne];
478
479 rama_s(0,p.rama_idx) += sens(1,ne)*(-output(0,ne)) + sens(0,ne)*output(1,ne);
480 rama_s(1,p.rama_idx) += sens(3,ne)*(-output(2,ne)) + sens(2,ne)*output(3,ne);
481 if(p.donor_idx !=-1) hbond_s(6,p.donor_idx ) += sens(4,ne);
482 if(p.acceptor_idx!=-1) hbond_s(6,p.acceptor_idx) += sens(5,ne);
483 }
484 }

489struct Conv1D : public CoordNode
490{
491 CoordNode& input;
492 int n_elem_input;
493 enum ActivationT {ReLU, Tanh, Identity};
494
495 int conv_width, in_channels, out_channels;
496
497 MatrixXf weights;
498 RowVectorXf bias;
499 ActivationT activation;
500
501 // temporaries
502 MatrixXf input_conv_format;
503 MatrixXf matmul_output;
504
505 Conv1D(hid_t grp, CoordNode& input_):
506 CoordNode (input_.n_elem - get_dset_size(3, grp, "weights")[0] + 1,
507 get_dset_size(3, grp, "weights")[2]),
508 input(input_),
509 n_elem_input(input.n_elem),
510
511 conv_width (get_dset_size(3, grp, "weights")[0]),
512 in_channels (get_dset_size(3, grp, "weights")[1]),
513 out_channels(get_dset_size(3, grp, "weights")[2]),
514
515 weights(conv_width*in_channels, out_channels),
516 bias(out_channels),
517 input_conv_format(n_elem, conv_width*in_channels),
518 matmul_output (n_elem, out_channels)
519 {
520 traverse_dset<3,float>(grp, "weights", [&](size_t nw, size_t in_c, size_t out_c, float x) {
521 weights(nw*in_channels+in_c, out_c) = x;});
522
523 traverse_dset<1,float>(grp, "bias", [&](size_t out_c, float x) {bias[out_c] = x;});
524
525 // Read activation
526 // This really should be a single string instead of a vector of them,
527 // but I currently only have a reader for string vectors
528 auto activation_str = read_attribute<vector<string>>(grp, ".", "activation");
529 if(activation_str.size() != 1u) throw string("Invalid number of activations");
530
531 if (activation_str[0] == "ReLU") activation = ReLU;
532 else if(activation_str[0] == "Tanh") activation = Tanh;
533 else if(activation_str[0] == "Identity") activation = Identity;
534 else throw string("Invalid activation name");
535 }
536
537 virtual void compute_value(ComputeMode mode) override {
538 Timer timer(string("conv1d"));
539 VecArray inputc = input.output;
540
541 int n_elem_output = n_elem;
542
543 // Copy input into proper format so that convolution becomes matmul
544 for(int nr=0; nr<n_elem_output; ++nr)
545 for(int nw=0; nw<conv_width; ++nw)
546 for(int nc=0; nc<in_channels; ++nc)
547 input_conv_format(nr,nw*in_channels+nc) = inputc(nc,nr+nw);
548
549 // Multiply by weights to make matmul_output
550 matmul_output.noalias() = input_conv_format * weights;
551 matmul_output.rowwise() += bias;
552
553 // Apply activation and store in output
554 // We also keep matmul_output around for storing information for backprop
555 switch(activation) {
556 case Identity:
557 for(int nr=0; nr<n_elem_output; ++nr)
558 for(int nc=0; nc<out_channels; ++nc)
559 output(nc,nr) = matmul_output(nr,nc);
560 break;
561 case ReLU:
562 for(int nr=0; nr<n_elem_output; ++nr)
563 for(int nc=0; nc<out_channels; ++nc)
564 output(nc,nr) = relu(matmul_output(nr,nc));
565 break;
566 case Tanh:
567 for(int nr=0; nr<n_elem_output; ++nr)
568 for(int nc=0; nc<out_channels; ++nc)
569 output(nc,nr) = tanh(matmul_output(nr,nc));
570 break;
571 }
572 }
573
574 virtual void propagate_deriv() override {
575 int n_elem_output = n_elem;
576
577 // Backpropagate activations
578 switch(activation) {
579 case Identity:
580 for(int nr=0; nr<n_elem_output; ++nr)
581 for(int nc=0; nc<out_channels; ++nc)
582 matmul_output(nr,nc) = sens(nc,nr);
583 break;
584 case ReLU:
585 for(int nr=0; nr<n_elem_output; ++nr)
586 for(int nc=0; nc<out_channels; ++nc)
587 matmul_output(nr,nc) = output(nc,nr)>0.f ? sens(nc,nr) : 0.f;
588 break;
589 case Tanh:
590 for(int nr=0; nr<n_elem_output; ++nr)
591 for(int nc=0; nc<out_channels; ++nc)
592 matmul_output(nr,nc) = sens(nc,nr) * (1.f-sqr(output(nc,nr)));
593 break;
594 }
595
596 // Backpropagate convolution
597 input_conv_format.noalias() = matmul_output * weights.transpose();
598
599 // Backpropagte into sens
600 VecArray inp_sens = input.sens;
601 for(int nr=0; nr<n_elem_output; ++nr)
602 for(int nw=0; nw<conv_width; ++nw)
603 for(int nc=0; nc<in_channels; ++nc)
604 inp_sens(nc,nr+nw) += input_conv_format(nr,nw*in_channels+nc);
605 }
606};
