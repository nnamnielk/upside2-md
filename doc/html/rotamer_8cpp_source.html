546 }
547
548 // Perform edge normalization for all edges
549 // We could perform it in the loop above, but it would insert a long dependency chain in the
550 // middle of the algorithm. The hope is that the processor will expose much more instruction
551 // parallelism in this loop. The loop process 2 edges at a time to fully utilize the horizontal
552 // adds.
553 for(int ne=0; ne<n_edge; ne+=2) {
554 auto cb11 = read4vec<w1>(cur_belief.x + ne*4*ws + 0);
555 auto cb12 = read4vec<w2>(cur_belief.x + ne*4*ws + 4*w1);
556 auto cb21 = read4vec<w1>(cur_belief.x + ne*4*ws + 4*ws);
557 auto cb22 = read4vec<w2>(cur_belief.x + ne*4*ws + 4*(ws+w1));
558
559 // let's approximately l1 normalize everything edges to avoid any numerical problems later
560 Float4 scales_for_unit_l1 = approx_rcp(horizontal_add(
561 horizontal_add(sum(cb11), sum(cb12)),
562 horizontal_add(sum(cb21), sum(cb22))));
563
564 store4vec<w1>(cur_belief.x + ne*4*ws + 0, cb11*scales_for_unit_l1.broadcast<0>());
565 store4vec<w2>(cur_belief.x + ne*4*ws + 4*w1, cb12*scales_for_unit_l1.broadcast<1>());
566 store4vec<w1>(cur_belief.x + ne*4*ws + 4*ws, cb21*scales_for_unit_l1.broadcast<2>());
567 store4vec<w2>(cur_belief.x + ne*4*ws + 4*(ws+w1), cb22*scales_for_unit_l1.broadcast<3>());
568 }
569 }
570};

607array<int,UPPER_ROT> calculate_n_elem(InteractionGraph<BT> &igraph) {
608 array<int,UPPER_ROT> result; // 0-rot is included
609 for(int& i: result) i=0;
610
611 unordered_map<unsigned,set<unsigned>> unique_ids;
612 for(int ne: range(igraph.n_elem1)) {
613 unsigned id = igraph.id1[ne];
614 unsigned selector = (1u<<n_bit_rotamer) - 1u;
615 unsigned rot = id & selector; id >>= n_bit_rotamer;
616 unsigned n_rot = id & selector; id >>= n_bit_rotamer;
617 if(rot>=n_rot) throw string("invalid rotamer number");
618 unique_ids[n_rot].insert(id);
619 }
620 for(auto& kv: unique_ids) {
621 if(kv.first >= UPPER_ROT) throw string("invalid rotamer count ")+to_string(kv.first);
622 result.at(kv.first) = kv.second.size();
623 }
624 return result;

854 void fill_holders()
855 {
856 Timer timer(std::string("rotamer_fill"));
857 edges11.reset();
858 for(int n_rot1: range(UPPER_ROT))
859 for(int n_rot2: range(UPPER_ROT))
860 if(edge_holders_matrix[n_rot1][n_rot2])
861 edge_holders_matrix[n_rot1][n_rot2]->reset();
862
863 for(int n_rot: range(UPPER_ROT))
864 if(node_holders_matrix[n_rot])
865 node_holders_matrix[n_rot]->reset();
866
867 vector<VecArray> energy_1body;
868 energy_1body.reserve(n_prob_nodes);
869 for(int i: range(n_prob_nodes))
870 energy_1body.emplace_back(prob_nodes[i]->output);
871
872 for(int n: range(igraph.n_elem1)) {
873 unsigned id = igraph.id1[n];
874 unsigned selector = (1u<<n_bit_rotamer) - 1u;
875 unsigned rot = id & selector; id >>= n_bit_rotamer;
876 unsigned n_rot = id & selector; id >>= n_bit_rotamer;
877 int index = igraph.loc1[n];
878
879 float energy = 0.f;
880 for(auto &a: energy_1body) energy += a(0,index);
881
882 // at this point, the node "prob" is really energy
883 node_holders_matrix[n_rot]->prob(rot,id) += energy;
884 }
885 for(int n_rot: range(1,UPPER_ROT))
886 if(node_holders_matrix[n_rot])
887 node_holders_matrix[n_rot]->convert_energy_to_prob(energy_cap, energy_cap_width);
888
889 // Fill edge probabilities
890 igraph.compute_edges();
891
892 const unsigned selector = (1u<<n_bit_rotamer) - 1u;
893 for(int ne=0; ne<igraph.n_edge; ++ne) {
894 int id1 = igraph.edge_id1[ne];
895 int id2 = igraph.edge_id2[ne];
896 float prob = expf(-igraph.edge_value[ne]); // value of edge is potential
897
898 if((id1&(selector<<n_bit_rotamer)) > (id2&(selector<<n_bit_rotamer))) swap(id1,id2);
899
900 unsigned rot1 = id1 & selector; id1 >>= n_bit_rotamer;
901 unsigned rot2 = id2 & selector; id2 >>= n_bit_rotamer;
902
903 unsigned n_rot1 = id1 & selector; id1 >>= n_bit_rotamer;
904 unsigned n_rot2 = id2 & selector; id2 >>= n_bit_rotamer;
905
906 edge_holders_matrix[n_rot1][n_rot2]->add_to_edge(ne, prob, id1, rot1, id2, rot2);
907 }
908
909 // for edges with a 1, we can just move it
910 for(int n_rot: range(2,UPPER_ROT))
911 if(edge_holders_matrix[1][n_rot])
912 edge_holders_matrix[1][n_rot]->move_edge_prob_to_node2();
