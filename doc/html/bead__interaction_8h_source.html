32 inline Float4 quadspline(
33 Vec<n_dim1,Float4> &d1, Vec<n_dim2,Float4> &d2,
34 const float inv_dtheta, const float inv_dx, const float* p[4],
35 const Vec<n_dim1,Float4> &x1, const Vec<n_dim2,Float4> &x2)
36 {
37 Float4 one(1.f);
38 auto displace = extract<0,3>(x2)-extract<0,3>(x1);
39 auto rvec1 = extract<3,6>(x1);
40 auto rvec2 = extract<3,6>(x2);
41
42 auto dist2 = mag2(displace);
43 auto inv_dist = rsqrt(dist2);
44 auto dist_coord = dist2*(inv_dist*Float4(inv_dx));
45 auto displace_unitvec = inv_dist*displace;
46
47 auto cos_cov_angle1 = dot(rvec1, displace_unitvec);
48 auto cos_cov_angle2 = dot(rvec2,-displace_unitvec);
49
50 // Spline evaluation
51 auto angular_sigmoid1 = deBoor_value_and_deriv(p, (cos_cov_angle1+one)*Float4(inv_dtheta)+one);
52 int o = n_knot_angular; const float* pp[4] = {p[0]+o, p[1]+o, p[2]+o, p[3]+o};
53
54 auto angular_sigmoid2 = deBoor_value_and_deriv(pp, (cos_cov_angle2+one)*Float4(inv_dtheta)+one);
55 o=n_knot_angular; pp[0]+=o; pp[1]+=o; pp[2]+=o; pp[3]+=o;
56
57 auto wide_cover = clamped_deBoor_value_and_deriv(pp, dist_coord, n_knot);
58 o=n_knot; pp[0]+=o; pp[1]+=o; pp[2]+=o; pp[3]+=o;
59
60 auto narrow_cover = clamped_deBoor_value_and_deriv(pp, dist_coord, n_knot);
61
62 // Partition derivatives
63 auto angular_weight = angular_sigmoid1.x() * angular_sigmoid2.x();
64
65 auto radial_deriv = Float4(inv_dx ) * (wide_cover.y() + angular_weight*narrow_cover.y());
66 auto angular_deriv1 = Float4(inv_dtheta) * angular_sigmoid1.y()*angular_sigmoid2.x()*narrow_cover.x();
67 auto angular_deriv2 = Float4(inv_dtheta) * angular_sigmoid1.x()*angular_sigmoid2.y()*narrow_cover.x();
68
69 auto rXX = angular_deriv1*rvec1 - angular_deriv2*rvec2;
70 auto deriv_dir = inv_dist * (rXX - dot(displace_unitvec,rXX)*displace_unitvec);
71
72 auto d_displace = radial_deriv * displace_unitvec + deriv_dir;
73 auto d_rvec1 = angular_deriv1 * displace_unitvec;
74 auto d_rvec2 = -angular_deriv2 * displace_unitvec;
75
76 auto coverage = wide_cover.x() + angular_weight*narrow_cover.x();
77
78 store<0,3>(d1, -d_displace);
79 store<3,6>(d1, d_rvec1);
80
81 store<0,3>(d2, d_displace);
82 store<3,6>(d2, d_rvec2);
83
84 return coverage;
85 }

135 struct PosDistSplineInteraction {
136 // spline-based distance interaction
137 // n_param is the number of basis splines (including those required to get zero
138 // derivative in the clamped spline)
139 // spline is constant over [0,dx] to avoid funniness at origin
140 // spline should be clamped at zero at the large end for cutoffs
141
142 constexpr static float inv_dx = 1.f/0.50f; // half-angstrom bins
143 constexpr static bool symmetric = true;
144 constexpr static int n_param=N_KNOT_SC_SC, n_dim1=3, n_dim2=3, simd_width=1; // 8 angstrom cutoff
145
146 static float cutoff(const float* p) {
147 return (n_param-2-1e-6)/inv_dx; // 1e-6 just insulates us from round-off error
148 }
149
150 static bool is_compatible(const float* p1, const float* p2) {
151 for(int i: range(n_param)) if(p1[i]!=p2[i]) return false;
152 return true;
153 }
154
155 static Int4 acceptable_id_pair(const Int4& id1, const Int4& id2) {
156 return id1.srl(n_bit_rotamer) != id2.srl(n_bit_rotamer);
157 }
158
159 static Float4 compute_edge(Vec<n_dim1,Float4> &d1, Vec<n_dim2,Float4> &d2, const float* p[4],
160 const Vec<n_dim1,Float4> &x1, const Vec<n_dim2,Float4> &x2) {
161 auto disp = x1-x2;
162 auto dist2 = mag2(disp);
163 auto inv_dist = rsqrt(dist2+Float4(1e-7f)); // 1e-7 is divergence protection
164 auto dist_coord = dist2*(inv_dist*inv_dx);
165
166 auto en = clamped_deBoor_value_and_deriv(p, dist_coord, n_param);
167 d1 = disp*(inv_dist*inv_dx*en.y());
168 d2 = -d1;
169 return en.x();
170 }
171
172 static void param_deriv(Vec<n_param> &d_param, const float* p,
173 const Vec<n_dim1> &x1, const Vec<n_dim2> &x2) {
174 auto dist_coord = inv_dx*mag(x1-x2);
175
176 int starting_bin;
177 float result[4];
178 clamped_deBoor_coeff_deriv(&starting_bin, result, dist_coord, n_param);
179 for(int i: range(4)) d_param[starting_bin+i] = result[i];
180 }
181 };
182}
