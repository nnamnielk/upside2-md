75std::vector<std::string> read_attribute<std::vector<std::string>>(hid_t h5, const char* path, const char* attr_name)
76try {
77 if(!h5_exists(h5, path))
78 throw std::string("path does not exist in h5 file");
79
80 auto attr = h5_obj(H5Aclose, H5Aopen_by_name(h5, path, attr_name, H5P_DEFAULT, H5P_DEFAULT));
81 auto space = h5_obj(H5Sclose, H5Aget_space(attr.get()));
82 auto dtype = h5_obj(H5Tclose, H5Aget_type (attr.get()));
83 if(H5Tis_variable_str(dtype.get())) throw std::string("variable-length strings not supported");
84
85 size_t maxchars = H5Tget_size(dtype.get());
86 if(maxchars==0) throw std::string("H5Tget_size error"); // defined error value
87
88 if(H5Sget_simple_extent_ndims(space.get()) != 1) throw std::string("wrong size for attribute");
89 hsize_t dims[1]; h5_noerr(H5Sget_simple_extent_dims(space.get(), dims, NULL));
90
91 auto tmp = std::unique_ptr<char[]>(new char[dims[0]*maxchars+1]);
92 std::fill(tmp.get(), tmp.get()+dims[0]*maxchars+1, '\0');
93 h5_noerr(H5Aread(attr.get(), dtype.get(), tmp.get()));
94
95 std::vector<std::string> ret;
96
97 auto g = [&](int i, char& s) {
98 std::string tmp(maxchars,'\0');
99 std::copy(&s, &s+maxchars, begin(tmp));
100 while(tmp.size() && tmp.back() == '\0') tmp.pop_back();
101 ret.push_back(tmp);
102 };
103 traverse_dataset_iteraction_helper<1,char,decltype(g)>()(tmp.get(), dims, g, maxchars);
104 return ret;
105} catch(const std::string &e) {
106 throw "while reading attribute '" + std::string(attr_name) + "' of '" + std::string(path) + "', " + e;
107}

183 const std::initializer_list<int> &chunk_dims,

198H5Obj create_earray(hid_t group, const char* name, hid_t dtype,
199 const std::vector<hsize_t>& dims_v, // any direction that is extendable must have dims == H5S_UNLIMITED
200 const std::vector<hsize_t>& chunk_dims_v,
201 bool compression_level) // 1 is often recommended
202{
203 if(dims_v.size() != chunk_dims_v.size()) throw std::string("invalid chunk dims");
204 std::vector<hsize_t> dims = dims_v;
205 std::vector<hsize_t> chunk_dims = chunk_dims_v;
206
207 hsize_t ndims = dims_v.size();
208 std::vector<hsize_t> max_dims_v(ndims);
209
210 for(size_t d=0; d<ndims; ++d) {
211 if(dims[d]==H5S_UNLIMITED) {
212 dims[d] = 0;
213 max_dims_v[d] = H5S_UNLIMITED;
214 } else {
215 max_dims_v[d] = dims[d];
216 }
217 if(chunk_dims[d]==0u) chunk_dims[d]=1;
218 }
219
220 auto space_id = h5_obj(H5Sclose, H5Screate_simple(ndims, dims.data(), max_dims_v.data()));
221
222 // setup chunked, possibly compressed storage
223 auto dcpl_id = h5_obj(H5Pclose, H5Pcreate(H5P_DATASET_CREATE));
224 h5_noerr(H5Pset_chunk(dcpl_id.get(), ndims, chunk_dims.data()));
225 h5_noerr(H5Pset_shuffle(dcpl_id.get())); // improves data compression
226 h5_noerr(H5Pset_fletcher32(dcpl_id.get())); // for verifying data integrity
227 if(compression_level) h5_noerr(H5Pset_deflate(dcpl_id.get(), compression_level));
228
229 return h5_obj(H5Dclose, H5Dcreate2(group, name, dtype, space_id.get(),
230 H5P_DEFAULT, dcpl_id.get(), H5P_DEFAULT));
231}

233void append_to_dset(hid_t dset, hid_t hdf_predtype, size_t n_new_data_elems, const void* new_data, int append_dim)
234try {
235 // Load current data size
236 auto space = h5_obj(H5Sclose, H5Dget_space(dset));
237 int ndims = h5_noerr(H5Sget_simple_extent_ndims(space.get()));
238 std::vector<hsize_t> dims(ndims);
239 h5_noerr("H5Sget_simple_extent_dims", H5Sget_simple_extent_dims(space.get(), dims.data(), NULL));
240
241 // Compute number of records in new_data
242 long int record_size = 1; // product of all dim sizes except the append_dim
243 for(int nd=0; nd<(int)dims.size(); ++nd) if(nd!=append_dim) record_size *= dims[nd];
244
245 long int n_records = n_new_data_elems / record_size;
246 if(n_records*record_size != (long)n_new_data_elems)
247 throw std::string("new data is not an integer number of records long");
248
249 // Enlarge dataset for new data
250 auto enlarged_dims = dims;
251 enlarged_dims[append_dim] += n_records;
252 h5_noerr("H5Dset_extent", H5Dset_extent(dset, enlarged_dims.data()));
253 auto enlarged_space = h5_obj(H5Sclose, H5Dget_space(dset));
254
255 // Dataspaces are required for the source and destination
256
257 // Source is just a simple dataspace
258 auto mem_dims = dims;
259 mem_dims[append_dim] = n_records;
260 auto mem_space = h5_obj(H5Sclose, H5Screate_simple(mem_dims.size(), mem_dims.data(), NULL));
261
262 // Destination is a hyperslab selection starting at the old length in the append direction
263 std::vector<hsize_t> starts(ndims);
264 for(int d=0; d<ndims; ++d) starts[d] = (d==append_dim) ? dims[d] : 0u;
265
266 h5_noerr("H5Sselect_hyperslab",
267 H5Sselect_hyperslab(enlarged_space.get(), H5S_SELECT_SET, starts.data(), nullptr, mem_dims.data(), nullptr));
268
269 // Perform write
270 h5_noerr(H5Dwrite(dset, hdf_predtype, mem_space.get(), enlarged_space.get(), H5P_DEFAULT, new_data));
271}
272catch(const std::string &e) {
273 throw "while appending to '" + std::string("a dataset") + "', " + e;
274}

277std::vector<std::string>
278node_names_in_group(const hid_t loc, const std::string grp_name) {
279 std::vector<std::string> names;
280
281 H5G_info_t info;
282 h5_noerr(H5Gget_info_by_name(loc, grp_name.c_str(), &info, H5P_DEFAULT));
283
284 for(int i=0; i<int(info.nlinks); ++i) {
285 // 1+ is for NULL terminator
286 size_t name_size = 1 + h5_noerr(
287 H5Lget_name_by_idx(loc, grp_name.c_str(), H5_INDEX_NAME, H5_ITER_INC, i,
288 nullptr, 0, H5P_DEFAULT));
289
290 auto tmp = std::unique_ptr<char[]>(new char[name_size]);
291 h5_noerr(H5Lget_name_by_idx(loc, grp_name.c_str(), H5_INDEX_NAME, H5_ITER_INC, i,
292 tmp.get(), name_size, H5P_DEFAULT));
293
294 names.emplace_back(tmp.get());
295 }
296 return names;
297}
