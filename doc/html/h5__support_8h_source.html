229void traverse_dset(hid_t group, const char* name, const F& f)
230try {
231 auto dset = h5_obj(H5Dclose, H5Dopen2(group, name, H5P_DEFAULT));
232 auto space = h5_obj(H5Sclose, H5Dget_space(dset.get()));
233
234 if(H5Sget_simple_extent_ndims(space.get()) != ndims)
235 throw std::string("wrong shape for array, expected " +
236 std::to_string(ndims) + " dimension(s) but got " +
237 std::to_string(H5Sget_simple_extent_ndims(space.get())) +
238 " dimension(s).");
239 hsize_t dims[ndims]; h5_noerr(H5Sget_simple_extent_dims(space.get(), dims, NULL));
240
241 size_t dim_product = 1;
242 for(int i=0; i<ndims; ++i) dim_product *= dims[i];
243
244 auto tmp = std::unique_ptr<T[]>(new T[dim_product]);
245
246 h5_noerr(H5Dread(dset.get(), select_predtype<T>(), H5S_ALL, H5S_ALL, H5P_DEFAULT, tmp.get()));
247 traverse_dataset_iteraction_helper<ndims,T,F>()(tmp.get(), dims, f);
248} catch(const std::string &e) {
249 throw "while traversing '" + std::string(name) + "', " + e;
250}
